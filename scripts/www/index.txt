            let ctx = null, unlocked = false;
            function init() {
                if(unlocked) return;
                const AC = window.AudioContext || window.webkitAudioContext;
                if(!AC) return; ctx = new AC();
                const o = ctx.createOscillator(); const g = ctx.createGain();
                g.gain.value = 0; o.connect(g); g.connect(ctx.destination);
                o.start(); o.stop(ctx.currentTime + 0.1); unlocked = true;
            }

            function createNodeTexture(node) {
                const res = 256; const canvas = document.createElement('canvas'); canvas.width = node.w * res; canvas.height = node.h * res;
                const ctx = canvas.getContext('2d'); const r = 0.2 * res; const w = canvas.width, h = canvas.height;
                if (node.type !== 'text' && node.type !== 'dashed_bg') {
                    ctx.beginPath(); ctx.moveTo(r, 4); ctx.lineTo(w - r, 4); ctx.quadraticCurveTo(w - 4, 4, w - 4, r);
                    ctx.lineTo(w - 4, h - r); ctx.quadraticCurveTo(w - 4, h - 4, w - r, h - 4);
                    ctx.lineTo(r, h - 4); ctx.quadraticCurveTo(4, h - 4, 4, h - r);
                    ctx.lineTo(4, r); ctx.quadraticCurveTo(4, 4, r, 4); ctx.closePath();
                }
                if (node.id === 'think' || node.type === 'solid') { ctx.fillStyle = '#2563eb'; ctx.fill(); ctx.strokeStyle = '#1d4ed8'; ctx.lineWidth = 6; ctx.stroke(); } 
                else if (node.type === 'dashed') { ctx.fillStyle = '#475569'; ctx.fill(); ctx.lineWidth = 5; ctx.setLineDash([12, 12]); ctx.strokeStyle = '#94a3b8'; ctx.stroke(); ctx.setLineDash([]); } 
                else if (node.type === 'dashed_bg') { 
                    ctx.fillStyle = 'rgba(71, 85, 105, 0.3)'; ctx.fillRect(0,0,w,h); 
                    ctx.lineWidth = 4; ctx.setLineDash([8, 8]); ctx.strokeStyle = '#64748b'; ctx.strokeRect(4,4,w-8,h-8); 
                    return new THREE.CanvasTexture(canvas); 
                } 
                else if (node.type === 'logic_green') { ctx.fillStyle = '#16a34a'; ctx.fill(); ctx.lineWidth = 4; ctx.strokeStyle = '#4ade80'; ctx.stroke(); }

                const mat = new THREE.PointsMaterial({ color: 0xffffff, size: 0.15, transparent: true, blending: THREE.AdditiveBlending });
                const sparks = new THREE.Points(geo, mat); sparks.position.copy(pos); sparks.position.z += 0.5; scene.add(sparks);
                const targets = [];
                for(let i=0; i<count; i++) {
                    const phi = Math.acos(-1 + (2 * i) / count), theta = Math.sqrt(count * Math.PI) * phi, r = 2.0 + Math.random();
                    targets.push(new THREE.Vector3(r * Math.cos(theta) * Math.sin(phi), r * Math.sin(theta) * Math.sin(phi), r * Math.cos(phi)));
                }
                const anim = { v: 0 };
                new TWEEN.Tween(anim).to({v: 1}, 800).onUpdate(() => {
                    const p = sparks.geometry.attributes.position.array;
                    for(let i=0; i<count; i++) { p[i*3]=targets[i].x*anim.v; p[i*3+1]=targets[i].y*anim.v; p[i*3+2]=targets[i].z*anim.v; }
                    sparks.geometry.attributes.position.needsUpdate = true; mat.opacity = 1 - anim.v;
                }).onComplete(() => scene.remove(sparks)).start();
            }

            function resetAll() {
                if (!isFocusMode) return; 
                isFocusMode = false; currentFocusedId = null;
                tooltipEl.style.opacity = 0;
                tooltipEl.style.pointerEvents = 'none';
                
                if(camTween) camTween.stop();
                if(targetTween) targetTween.stop();

                const currentTarget = controls.target;
                camTween = new TWEEN.Tween(camera.position).to({ x: currentTarget.x, y: currentTarget.y, z: 65 }, 800).easing(TWEEN.Easing.Cubic.InOut).start();
                targetTween = new TWEEN.Tween(controls.target).to({ x: currentTarget.x, y: currentTarget.y, z: 0 }, 800).easing(TWEEN.Easing.Cubic.InOut).onUpdate(() => controls.update()).start();
                
                resetTargets.forEach(obj => {
                    if (obj.material) {
                        const mats = Array.isArray(obj.material) ? obj.material : [obj.material];
                        mats.forEach(m => {
                            if(m.opTween) m.opTween.stop(); 
                            const targetOp = (obj.userData && obj.userData.type === 'dashed_bg') ? 0.6 : (m.emissiveIntensity > 0 ? 1.0 : 0.6);
                            m.opTween = new TWEEN.Tween(m).to({ opacity: targetOp }, 600).start();
                        });
                    }
                });
            }

            container.addEventListener('pointerdown', (e) => {
                if(e.target.tagName==='BUTTON') return;
                tX = e.clientX; tY = e.clientY; isD = false;
                mouse.x = (e.clientX/window.innerWidth)*2-1; mouse.y = -(e.clientY/window.innerHeight)*2+1;
                ray.setFromCamera(mouse, camera); const inter = ray.intersectObjects(interactableNodes);
                if (inter.length > 0) { 
                    hoveredNode = inter[0].object; hoveredNode.userData.isHovered = true; 
                    triggerHaptic('light'); 
                }
            });

            container.addEventListener('pointermove', (e) => {
                if(e.pointerType === 'mouse') {
                    mouse.x = (e.clientX/window.innerWidth)*2-1; mouse.y = -(e.clientY/window.innerHeight)*2+1;
                    ray.setFromCamera(mouse, camera); const inter = ray.intersectObjects(interactableNodes);
                    if (inter.length > 0) { 
                        if(hoveredNode && hoveredNode !== inter[0].object) hoveredNode.userData.isHovered = false;
                        if(!hoveredNode || hoveredNode !== inter[0].object) AudioEngine.playHover();
                        hoveredNode = inter[0].object; hoveredNode.userData.isHovered = true; container.style.cursor = 'pointer';
                    } else { if(hoveredNode) hoveredNode.userData.isHovered = false; hoveredNode = null; container.style.cursor = 'default'; }
                } else if (Math.abs(e.clientX-tX)>8 || Math.abs(e.clientY-tY)>8) { isD = true; if(hoveredNode){hoveredNode.userData.isHovered=false;hoveredNode=null;} }
            });

            container.addEventListener('pointerup', (e) => {
                if(e.target.tagName==='BUTTON') return;
                if (isD) return; 

                mouse.x = (e.clientX/window.innerWidth)*2-1; mouse.y = -(e.clientY/window.innerHeight)*2+1;
                ray.setFromCamera(mouse, camera);
                const inter = ray.intersectObjects(interactableNodes);

                if (inter.length > 0) { 
                    const clickedNode = inter[0].object;
                    if (currentFocusedId !== clickedNode.userData.id) {
                        focusOnNode(clickedNode); 
                        setTimeout(() => showTooltip(clickedNode, window.innerWidth / 2, window.innerHeight / 2 + 100), 100); 
                    } else {
                        resetAll(); 
                    }
                } else {
                    if (isFocusMode) resetAll(); 
                }

                if (e.pointerType !== 'mouse' && hoveredNode) { hoveredNode.userData.isHovered = false; hoveredNode = null; }
            });

            function focusOnNode(node) {
                isFocusMode = true; currentFocusedId = node.userData.id;
                triggerHaptic('heavy'); AudioEngine.playFocus(); createRipple(node); createSparks(node.position);
                
                if(camTween) camTween.stop();
                if(targetTween) targetTween.stop();

                resetTargets.forEach(obj => {
                    if (obj.material) {
                        const mats = Array.isArray(obj.material) ? obj.material : [obj.material];
                        mats.forEach(m => {
                            if(m.opTween) m.opTween.stop();
                            m.opTween = new TWEEN.Tween(m).to({ opacity: (obj === node) ? 1.0 : 0.15 }, 500).start();
                        });
                    }
                });

                camTween = new TWEEN.Tween(camera.position).to({ x: node.position.x, y: node.position.y, z: 40 }, 1000).easing(TWEEN.Easing.Cubic.Out).start();
                targetTween = new TWEEN.Tween(controls.target).to({ x: node.position.x, y: node.position.y, z: 0 }, 1000).easing(TWEEN.Easing.Cubic.Out).onUpdate(() => controls.update()).start();
            }

            };

            const clock = new THREE.Clock();
            function animate() {
                requestAnimationFrame(animate);
                const t = clock.getElapsedTime();
                pulseElectrons.forEach(e => { e.userData.progress = (e.userData.progress + e.userData.speed) % 1.0; e.position.copy(e.userData.curve.getPointAt(e.userData.progress * e.userData.limit)); });
                interactableNodes.forEach(mesh => {
                    const d = mesh.userData, s = d.isHovered ? 1.12 : 1.0;
                    mesh.scale.x += (s - mesh.scale.x) * 0.15; mesh.scale.y += (s - mesh.scale.y) * 0.15; mesh.scale.z += (s - mesh.scale.z) * 0.15;
                    
                    if (!d.isText && mesh.material instanceof Array) {
                        const targetEmissive = d.isHovered ? 0.3 : 0.0;
                        mesh.material[4].emissiveIntensity += (targetEmissive - mesh.material[4].emissiveIntensity) * 0.1;
                        mesh.material[4].emissive.setHex(0xffffff);
                    }
                    mesh.position.z = Math.sin(t * 2 + d.randomPhase) * 0.12;
                });
                radarMat.uniforms.uTime.value = t;
                stars.rotation.y = t * 0.01;
                TWEEN.update(); controls.update(); composer.render(); 
            }
            animate();

                    const loader = document.getElementById('loading-screen');
                    if (loader) {

                        setTimeout(() => loader.remove(), 600); // зӯүеҫ…ж·ЎеҮәеҠЁз”»з»“жқҹеҶҚз§»йҷӨ DOM
                    }
                }, isMobileScreen ? 800 : 400); // еҰӮжһңжҳҜжүӢжңәпјҢеӨҡз»ҷ 0.4 з§’и®©еӨҚжқӮзҡ„зқҖиүІеҷЁе®ҢжҲҗйў„зғӯ
            });

        })();
