        function gotoPage(page) {
            let currentPath = window.location.pathname;
            let baseDir = currentPath.substring(0, currentPath.lastIndexOf('/') + 1);
            let targetUrl = baseDir + page;

            debugDiv.style.position = 'fixed';
            debugDiv.style.bottom = '100px';
            debugDiv.style.left = '10px';
            debugDiv.style.background = 'rgba(0,0,0,0.7)';
            debugDiv.style.color = 'white';
            debugDiv.style.padding = '8px';
            debugDiv.style.borderRadius = '8px';
            debugDiv.style.zIndex = '9999';
            debugDiv.style.fontSize = '12px';
            debugDiv.innerText = '跳转到: ' + targetUrl;
            document.body.appendChild(debugDiv);
            setTimeout(() => debugDiv.remove(), 1500);
            
            window.location.href = targetUrl;
        }

            function createNodeTexture(node) {
                const res = 256;
                const canvas = document.createElement('canvas');
                canvas.width = node.w * res;
                canvas.height = node.h * res;
                const ctx = canvas.getContext('2d');
                const r = 0.2 * res;
                const w = canvas.width, h = canvas.height;
                if (node.type !== 'text') {
                    ctx.beginPath();
                    ctx.moveTo(r, 4);
                    ctx.lineTo(w - r, 4);
                    ctx.quadraticCurveTo(w - 4, 4, w - 4, r);
                    ctx.lineTo(w - 4, h - r);
                    ctx.quadraticCurveTo(w - 4, h - 4, w - r, h - 4);
                    ctx.lineTo(r, h - 4);
                    ctx.quadraticCurveTo(4, h - 4, 4, h - r);
                    ctx.lineTo(4, r);
                    ctx.quadraticCurveTo(4, 4, r, 4);
                    ctx.closePath();
                }
                if (node.id === 'elec_think' || node.id === 'galv_think' || node.type === 'solid') {
                    ctx.fillStyle = '#4B74C2';
                    ctx.fill();
                    ctx.strokeStyle = '#2B4A87';
                    ctx.lineWidth = 6;
                    ctx.stroke();
                    ctx.fillStyle = '#FFF';
                } else if (node.type === 'dashed') {
                    ctx.fillStyle = '#FFF';
                    ctx.fill();
                    ctx.lineWidth = 5;
                    ctx.setLineDash([15, 15]);
                    ctx.strokeStyle = '#222';
                    ctx.stroke();
                    ctx.setLineDash([]);
                    ctx.fillStyle = '#000';
                } else if (node.type === 'logic_green') {
                    ctx.fillStyle = '#F4FAEB';
                    ctx.fill();
                    ctx.lineWidth = 4;
                    ctx.strokeStyle = '#5E7A8A';
                    ctx.stroke();
                    ctx.fillStyle = '#333';
                } else if (node.type === 'logic_pink') {
                    ctx.fillStyle = '#FDF2F4';
                    ctx.fill();
                    ctx.lineWidth = 4;
                    ctx.strokeStyle = '#5E7A8A';
                    ctx.stroke();
                    ctx.fillStyle = '#333';
                } else {
                    ctx.fillStyle = '#000';
                }

                let fontSize = Math.min(0.45 * res, (w - 40) / (node.text.length));
                ctx.font = `bold ${fontSize}px sans-serif`;
                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle';
                ctx.fillText(node.text, w / 2, h / 2 + 0.05 * res);
                const texture = new THREE.CanvasTexture(canvas);
                texture.anisotropy = renderer.capabilities.getMaxAnisotropy();
                return texture;
            }

            nodes.forEach(node => {
                const texture = createNodeTexture(node);
                let mesh;
                if (node.type === 'text') {
                    mesh = new THREE.Mesh(new THREE.PlaneGeometry(node.w, node.h), new THREE.MeshBasicMaterial({ map: texture, transparent: true }));
                    mesh.position.set(node.x, node.y, 0.2);
                } else {
                    const geo = new THREE.BoxGeometry(node.w, node.h, 0.2);
                    const matS = new THREE.MeshStandardMaterial({ color: (node.type === 'solid' || node.id.includes('think')) ? 0x4B74C2 : (node.type === 'logic_green' ? 0xF4FAEB : (node.type === 'logic_pink' ? 0xFDF2F4 : 0xFFFFFF)), roughness: 0.6, transparent: true });
                    const matF = new THREE.MeshStandardMaterial({ map: texture, roughness: 0.6, transparent: true });
                    mesh = new THREE.Mesh(geo, [matS, matS, matS, matS, matF, matF]);
                    mesh.position.set(node.x, node.y, 0);
                }
                mesh.userData = { id: node.id, baseZ: mesh.position.z, randomPhase: Math.random() * Math.PI * 2, isHovered: false, isText: node.type === 'text' };
                interactableNodes.push(mesh);
                scene.add(mesh);

            rawPaths.forEach(p => {
                const pts = p.points.map(v => new THREE.Vector3(...v));
                let cDist = 0;
                for (let i = 0; i < pts.length - 1; i++) {
                    const start = pts[i], end = pts[i + 1], segLen = start.distanceTo(end);
                    if (i > 0) {
                        const s = new THREE.Mesh(new THREE.SphereGeometry(0.05, 8, 8), new THREE.MeshStandardMaterial({ color: p.color }));
                        s.position.copy(start);
                        s.position.z = -0.1;
                        scene.add(s);
                    }
                    const dir = new THREE.Vector3().subVectors(end, start).normalize();
                    const lineMat = new THREE.MeshStandardMaterial({ color: p.color, roughness: 0.7, emissive: p.color, emissiveIntensity: 0, transparent: true });

                    if (i === pts.length - 2) {
                        const cone = new THREE.Mesh(new THREE.ConeGeometry(0.15, 0.5, 16), lineMat);
                        cone.position.copy(new THREE.Vector3().copy(end).addScaledVector(dir, -0.25));
                        cone.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), dir);
                        cone.position.z = -0.1;
                        cone.userData.dist = cDist + segLen;
                        pulseLines.push(cone);
                        scene.add(cone);
                        if (segLen > 0.5) {
                            const line = new THREE.Mesh(new THREE.CylinderGeometry(0.05, 0.05, segLen - 0.5, 8), lineMat);
                            line.position.copy(new THREE.Vector3().copy(start).addScaledVector(dir, (segLen - 0.5) / 2));
                            line.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), dir);
                            line.position.z = -0.1;
                            line.userData.dist = cDist + (segLen - 0.5) / 2;
                            pulseLines.push(line);
                            scene.add(line);
                        }
                    } else {
                        const line = new THREE.Mesh(new THREE.CylinderGeometry(0.05, 0.05, segLen, 8), lineMat);
                        line.position.copy(new THREE.Vector3().copy(start).addScaledVector(dir, segLen / 2));
                        line.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), dir);
                        line.position.z = -0.1;
                        line.userData.dist = cDist + segLen / 2;
                        pulseLines.push(line);
                        scene.add(line);
                    }
                    cDist += segLen;
                }
            });

            function showTooltip(node, x, y) {
                const data = tooltipData[node.userData.id];
                if (!data) return;
                let hintText = '';
                // 根据节点id显示不同的提示
                if (node.userData.id === 'plate') hintText = '✨ 双击进入电镀铜虚拟实验';
                else if (node.userData.id === 'li') hintText = '🔋 双击查看锂电池3D演示';
                else if (['elec_cell', 'galv_cell', 'et3', 'refine', 'metal', 'chlor', 'lead', 'zn', 'fuel'].includes(node.userData.id)) hintText = '👉 双击跳转至详细页面';
                else hintText = '👉 双击聚焦节点';
                tooltipEl.innerHTML = `<div class="tooltip-title">${data.title}</div><div class="tooltip-desc">${data.desc}</div><div class="tooltip-eq">${data.eq}</div><div class="tooltip-hint">${hintText}</div>`;
                let px = x, py = y - 60;
                if (px < 150) px = 150;
                if (px > window.innerWidth - 150) px = window.innerWidth - 150;
                if (py < 100) py = y + 80;
                tooltipEl.style.left = `${px}px`;
                tooltipEl.style.top = `${py}px`;
                tooltipEl.style.opacity = 1;
                interactableNodes.forEach(n => n.userData.isHovered = (n === node));
            }

            function hideTooltip() {
                tooltipEl.style.opacity = 0;
                interactableNodes.forEach(n => n.userData.isHovered = false);
            }

            function focusOnNode(node) {
                isFocusMode = true;
                hideTooltip();
                if (!camera.userData.origPos) {
                    camera.userData.origPos = camera.position.clone();
                    camera.userData.origTarget = controls.target.clone();
                }
                interactableNodes.forEach(n => {
                    const op = n === node ? 1 : 0.15;
                    if (n.material instanceof Array) {
                        n.material.forEach(m => new TWEEN.Tween(m).to({ opacity: op }, 800).start());
                    } else {
                        new TWEEN.Tween(n.material).to({ opacity: op }, 800).start();
                    }
                });
                pulseLines.forEach(l => new TWEEN.Tween(l.material).to({ opacity: 0.1, emissiveIntensity: 0 }, 800).start());
                new TWEEN.Tween(camera.position).to(new THREE.Vector3(node.position.x, node.position.y, node.position.z + 15), 1000).easing(TWEEN.Easing.Cubic.InOut).start();
                new TWEEN.Tween(controls.target).to({ x: node.position.x, y: node.position.y, z: node.position.z }, 1000).easing(TWEEN.Easing.Cubic.InOut).onUpdate(() => controls.update()).start();
            }

            // 添加退出聚焦模式（点击空白区域）
            container.addEventListener('click', (e) => {
                if (!isFocusMode) return;
                mouse.x = (e.clientX / window.innerWidth) * 2 - 1;
                mouse.y = -(e.clientY / window.innerHeight) * 2 + 1;
                raycaster.setFromCamera(mouse, camera);
                const intersects = raycaster.intersectObjects(interactableNodes);
                if (intersects.length === 0) {
                    isFocusMode = false;
                    interactableNodes.forEach(n => {
                        if (n.material instanceof Array) {
                            n.material.forEach(m => new TWEEN.Tween(m).to({ opacity: 1 }, 800).start());
                        } else {
                            new TWEEN.Tween(n.material).to({ opacity: 1 }, 800).start();
                        }
                    });
                    pulseLines.forEach(l => new TWEEN.Tween(l.material).to({ opacity: 1 }, 800).start());
                    if (camera.userData.origPos) {
                        new TWEEN.Tween(camera.position).to(camera.userData.origPos, 1000).easing(TWEEN.Easing.Cubic.InOut).start();
                        new TWEEN.Tween(controls.target).to(camera.userData.origTarget, 1000).easing(TWEEN.Easing.Cubic.InOut).onUpdate(() => controls.update()).start();
                    }
                }
            });

            // 节点点击事件（支持双击跳转/聚焦）
            container.addEventListener('pointerdown', (e) => {
                mouse.x = (e.clientX / window.innerWidth) * 2 - 1;
                mouse.y = -(e.clientY / window.innerHeight) * 2 + 1;
                raycaster.setFromCamera(mouse, camera);
                const intersects = raycaster.intersectObjects(interactableNodes);
                const now = Date.now();
                if (now - lastTapTime < 350) {
                    clearTimeout(tapTimeout);
                    if (intersects.length > 0) {
                        const node = intersects[0].object;
                        const nodeId = node.userData.id;

                        // --- 定义跳转映射表（节点ID -> 页面文件名）---
                        const pageMap = {
                            'elec_cell': '电解池模型.html',
                            'galv_cell': '原电池模型.html',
                            'et3': '放电顺序.html',
                            'plate': '电镀铜.html',
                            'refine': '电解精炼.html',
                            'metal': '电解氧化铝.html',
                            'chlor': '氯碱工业.html',
                            'lead': '铅蓄电池.html',
                            'li': '锂离子电池.html',
                            'zn': '锌锰干电池.html',
                            'fuel': '氢氧燃料电池.html'
                        };

                        if (pageMap[nodeId]) {
                            // 跳转到对应的子页面
                            hideTooltip();
                            gotoPage(pageMap[nodeId]);
                        } else {
                            // 其他节点执行聚焦动画
                            focusOnNode(node);
                        }
                    }
                } else {
                    if (intersects.length > 0) {
                        const node = intersects[0].object;
                        tapTimeout = setTimeout(() => showTooltip(node, e.clientX, e.clientY), 350);
                    } else {
                        hideTooltip();
                    }
                }
                lastTapTime = now;
            });

            App.resizeHome = function () {
                camera.aspect = window.innerWidth / window.innerHeight;
                camera.updateProjectionMatrix();
                renderer.setSize(window.innerWidth, window.innerHeight);
            };

            const clock = new THREE.Clock();
            function animate() {
                requestAnimationFrame(animate);
                const time = clock.getElapsedTime();

                pulseLines.forEach(item => {
                    if (!isFocusMode) {
                        item.material.emissiveIntensity = Math.pow(Math.max(0, Math.sin(time * 4.0 - item.userData.dist * 0.8)), 2) * 1.8;
                    }
                });
                interactableNodes.forEach(mesh => {
                    const d = mesh.userData;
                    const scale = d.isHovered ? 1.15 : 1.0;
                    mesh.scale.lerp(new THREE.Vector3(scale, scale, scale), 0.1);
                    mesh.position.z = d.baseZ + Math.sin(time * 2 + d.randomPhase) * 0.15;
                    if (!d.isText && mesh.material instanceof Array) {
                        const tc = (d.isHovered && !isFocusMode) ? new THREE.Color(0x224477) : new THREE.Color(0x000000);
                        mesh.material[0].emissive.lerp(tc, 0.1);
                        mesh.material[4].emissive.lerp(tc, 0.1);
                    }
                });
                particlesMesh.rotation.z = time * 0.02;
                particlesMesh.rotation.y = Math.sin(time * 0.05) * 0.1;
                TWEEN.update();
                controls.update();
                renderer.render(scene, camera);
            }
            animate();
        })();

            const tip = document.getElementById('orientation-tip');
            if (!tip) return;
            if (window.innerHeight > window.innerWidth) {
                tip.style.display = 'flex';
            } else {
                tip.style.display = 'none';
            }
        }
        window.addEventListener('resize', checkOrientation);
        window.addEventListener('orientationchange', checkOrientation);
