        // ========== 返回主页函数 ==========

            debugDiv.style.color = 'white';
            debugDiv.style.padding = '8px';
            debugDiv.style.borderRadius = '8px';
            debugDiv.style.zIndex = '9999';
            debugDiv.style.fontSize = '12px';
            debugDiv.innerText = '返回主页: ' + targetUrl;
            document.body.appendChild(debugDiv);
            setTimeout(() => debugDiv.remove(), 1500);
            window.location.href = targetUrl;
        }

        document.getElementById('back-to-home').addEventListener('click', goBackHome);

        // ========== 原有代码（未作任何功能改动） ==========
        let camera, scene, renderer, controls;
        let diagramGroup;
        const ionObjects = []; // 存储离子对象以实现悬浮动画

        init();
        animate();

        function init() {
            const container = document.getElementById('canvas-container');

            // 1. 初始化场景和相机
            scene = new THREE.Scene();
            
            // 视角拉远以便看清全貌
            camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 1, 5000);
            camera.position.set(0, 0, 1800); 

            // 2. 初始化 CSS3D 渲染器
            renderer = new THREE.CSS3DRenderer();
            renderer.setSize(window.innerWidth, window.innerHeight);
            container.appendChild(renderer.domElement);

            // 3. 初始化控制器
            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;
            controls.maxDistance = 3500;
            controls.minDistance = 600;

            // --- 4. 构建 3D 图表组 ---
            diagramGroup = new THREE.Group();
            scene.add(diagramGroup);

            // 背景板 (分为左右，Z 轴稍微靠后)
            createCSS3DObject('<div class="plane-left"></div>', -400, 0, -50, diagramGroup);
            createCSS3DObject('<div class="plane-right"></div>', 400, 0, -50, diagramGroup);

            // 坐标轴 (Z = 0)
            createCSS3DObject('<div class="axis-x"></div>', 0, 0, 0, diagramGroup);
            createCSS3DObject('<div class="axis-y"></div>', 0, 0, 0, diagramGroup);

            // 干扰离子高亮背板 (放置在水放电区 H+ 和 OH- 的正后方，Z = 15，位置略微偏下强调底部)
            createCSS3DObject('<div class="highlight-column"></div>', 50, -20, 15, diagramGroup);

            // 大标签和区域文字 (Z = 20，悬浮在背景和轴线前)
            createCSS3DObject('<div class="label-huge">强</div>', -650, 220, 20, diagramGroup);
            createCSS3DObject('<div class="label-huge">弱</div>', 650, 220, 20, diagramGroup);
            
            // 将“（酸）”分离出来，并将其 x 坐标设置为 -315，精准对齐下方左侧的 H+ 微粒
            createCSS3DObject('<div class="label-text">自身放电</div>', -400, 180, 20, diagramGroup);
            createCSS3DObject('<div class="label-text">（酸）</div>', -315, 110, 20, diagramGroup);
            
            createCSS3DObject('<div class="label-text">水放电</div>', 400, 160, 20, diagramGroup);
            
            createCSS3DObject('<div class="label-text">（水）</div>', 40, 260, 20, diagramGroup);
            createCSS3DObject('<div class="label-text">（水）</div>', 40, -260, 20, diagramGroup);
            
            createCSS3DObject('<div class="label-desc">阳离子氧化性</div>', 700, 30, 20, diagramGroup);
            createCSS3DObject('<div class="label-desc">阴离子还原性</div>', 700, -30, 20, diagramGroup);

            // --- 5. 渲染离子微粒序列 ---
            const yCation = 40;   // 阳离子 Y 坐标
            const yAnion = -40;   // 阴离子 Y 坐标
            const zIons = 30;     // 离子的基础 Z 深度

            // 数据定义 (包含文字 t、X 轴位置 x、放电产物 p)
            const cationsData = [
                { t: 'Ag<sup>+</sup>', x: -700, p: 'Ag' }, { t: '>', x: -635 },
                { t: 'Fe<sup>3+</sup>', x: -570, p: 'Fe²⁺' }, { t: '>', x: -500 },
                { t: 'Cu<sup>2+</sup>', x: -440, p: 'Cu' }, { t: '>', x: -375 },
                { t: 'H<sup>+</sup>', x: -315, p: 'H₂↑' }, { t: '>', x: -260 },
                { t: 'Fe<sup>2+</sup>', x: -200, p: 'Fe' }, { t: '>', x: -130 },
                { t: 'Zn<sup>2+</sup>', x: -60, p: 'Zn' }, { t: '>', x: 0 },
                { t: 'H<sup>+</sup>', x: 50, p: 'H₂↑' }, { t: '>', x: 100 },
                { t: 'Al<sup>3+</sup>', x: 160, p: 'H₂↑ (水放电)' }, { t: '>', x: 220 },
                { t: 'Mg<sup>2+</sup>', x: 280, p: 'H₂↑ (水放电)' }, { t: '>', x: 345 },
                { t: 'Na<sup>+</sup>', x: 405, p: 'H₂↑ (水放电)' }, { t: '>', x: 465 },
                { t: 'Ca<sup>2+</sup>', x: 525, p: 'H₂↑ (水放电)' }, { t: '>', x: 590 },
                { t: 'K<sup>+</sup>', x: 650, p: 'H₂↑ (水放电)' }
            ];

            const anionsData = [
                { t: '活泼电极', x: -660, p: '金属阳离子' }, { t: '>', x: -555 },
                { t: 'S<sup>2-</sup>', x: -480, p: 'S↓' }, { t: '>', x: -415 },
                { t: 'I<sup>-</sup>', x: -355, p: 'I₂' }, { t: '>', x: -300 },
                { t: 'Br<sup>-</sup>', x: -240, p: 'Br₂' }, { t: '>', x: -180 },
                { t: 'Cl<sup>-</sup>', x: -120, p: 'Cl₂↑' }, { t: '>', x: -50 },
                { t: 'OH<sup>-</sup>', x: 50, p: 'O₂↑ + H₂O' }, { t: '>', x: 110 },
                { t: '最高价含氧酸根', x: 250, p: 'O₂↑ (水放电)' }, { t: '>', x: 395 },
                { t: 'F<sup>-</sup>', x: 460, p: 'O₂↑ (水放电)' }
            ];

            // 遍历并生成 3D 对象
            cationsData.forEach(item => {
                const isArrow = item.t === '>';
                const className = isArrow ? 'arrow' : 'ion-card';
                // 让左边（强）的离子在 Z 轴上稍微凸出，右边（弱）的稍微靠后，增强 3D 感
                const zOffset = isArrow ? 0 : (-item.x * 0.05); 
                // 添加 data-product 属性
                const productAttr = item.p ? `data-product="产物: ${item.p}"` : '';
                const obj = createCSS3DObject(`<div class="${className}" ${productAttr}>${item.t}</div>`, item.x, yCation, zIons + zOffset, diagramGroup);
                if (!isArrow) {
                    obj.userData = { baseY: yCation, randomPhase: Math.random() * Math.PI * 2 };
                    ionObjects.push(obj);
                }
            });

            anionsData.forEach(item => {
                const isArrow = item.t === '>';
                const className = isArrow ? 'arrow' : 'ion-card';
                const zOffset = isArrow ? 0 : (-item.x * 0.05);
                // 添加 data-product 属性
                const productAttr = item.p ? `data-product="产物: ${item.p}"` : '';
                const obj = createCSS3DObject(`<div class="${className}" ${productAttr}>${item.t}</div>`, item.x, yAnion, zIons + zOffset, diagramGroup);
                if (!isArrow) {
                    obj.userData = { baseY: yAnion, randomPhase: Math.random() * Math.PI * 2 };
                    ionObjects.push(obj);
                }
            });

            // 初始化角度，微调视角
            // 修正：大幅减小了默认的倾斜和旋转角度，消除“向左下方斜”的视觉错觉
            diagramGroup.rotation.y = 0.10;
            diagramGroup.rotation.x = 0.02;
            diagramGroup.rotation.z = 0;

            window.addEventListener('resize', onWindowResize, false);
        }

        // 辅助函数：创建 CSS3D 对象
        function createCSS3DObject(html, x, y, z, parent) {
            const div = document.createElement('div');
            // 为了保证交互效果，使用一个透明 wrapper 包裹实际元素
            div.style.transformStyle = 'preserve-3d'; 
            div.innerHTML = html;
            
            const object = new THREE.CSS3DObject(div);
            object.position.set(x, y, z);
            parent.add(object);
            return object;
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        function animate() {
            requestAnimationFrame(animate);

            // 增加微粒的上下悬浮动画效果
            const time = Date.now() * 0.002;
            ionObjects.forEach((obj) => {
                obj.position.y = obj.userData.baseY + Math.sin(time + obj.userData.randomPhase) * 6;
            });

            controls.update();
            renderer.render(scene, camera);
        }
