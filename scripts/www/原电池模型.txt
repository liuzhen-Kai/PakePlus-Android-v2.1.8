        function generateTextTexture(text, color, fontSize = 60) {
            const canvas = document.createElement('canvas'); const context = canvas.getContext('2d');
            context.font = `bold ${fontSize}px "Microsoft YaHei", sans-serif`;
            const lines = text.split('\n'); let maxWidth = 0;
            lines.forEach(line => { const w = context.measureText(line).width; if (w > maxWidth) maxWidth = w; });
            const lineHeight = fontSize * 1.2; canvas.width = maxWidth + 20; canvas.height = lines.length * lineHeight + 20;
            if (text.trim() === "") { canvas.width = 2; canvas.height = 2; }
            context.font = `bold ${fontSize}px "Microsoft YaHei", sans-serif`; context.fillStyle = (color === '#000000') ? '#ffffff' : color; 
            context.textAlign = 'center'; context.textBaseline = 'middle';
            const startY = canvas.height / 2 - ((lines.length - 1) * lineHeight) / 2;
            if (text.trim() !== "") lines.forEach((line, index) => { context.fillText(line, canvas.width / 2, startY + index * lineHeight); });
            const texture = new THREE.CanvasTexture(canvas); texture.needsUpdate = true;
            return { texture, width: canvas.width, height: canvas.height };
        }

        function applyGlowAnimation(sprite, isGlowing, time) {
            if(!sprite || sprite.scale.x < 0.01) return; 
            if (isGlowing) {
                const flashScale = 1.0 + Math.sin(time * 0.005) * 0.15; sprite.scale.copy(sprite.userData.baseScale).multiplyScalar(flashScale);
                if (!sprite.userData.baseColor) sprite.userData.baseColor = sprite.material.color.clone();
                sprite.material.color.copy(sprite.userData.baseColor).multiplyScalar(2.0); 
            } else {
                sprite.scale.lerp(sprite.userData.baseScale, 0.15); if (sprite.userData.baseColor) sprite.material.color.copy(sprite.userData.baseColor);
            }
        }

        function animate(time) {
            requestAnimationFrame(animate); TWEEN.update(time); 

            // HUD 仪表盘波动
            if (animState.isCircuitClosed) {
                const fluctuation = Math.sin(time * 0.01) * 0.05 + Math.cos(time * 0.003) * 0.02;
                const currentVal = (animState.baseCurrent + fluctuation).toFixed(2);
                meterText.innerText = `I: ${currentVal}A`;
                const angle = -60 + (currentVal / 2.0) * 120;
                meterNeedle.style.transform = `translateX(-50%) rotate(${angle}deg)`;
            }

            let isHoveringNode = false;
            let forceHighlightElectron = false, forceHighlightIon = false, forceG1 = false, forceG2 = false;

            if (!animState.isTouring) {
                raycaster.setFromCamera(mouse, camera); const intersects = raycaster.intersectObjects(interactableObjects);
                animState.redCube = 1.0; animState.blueCube = 1.0; animState.greenCubeL = 1.0; animState.greenCubeR = 1.0;
                let tooltipText = ""; let tooltipPos = ""; let borderColor = "#3b82f6";

                if (intersects.length > 0 && mouse.x !== -9999) {
                    const name = intersects[0].object.userData.name; document.body.style.cursor = 'pointer'; isHoveringNode = true;
                    
                    // 【核心互动：盲盒揭晓】
                    if (animState.isTestMode && testModeMapping[name]) {
                        tooltipText = "<div style='color:#e879f9;font-size:11px;margin-bottom:3px;'>💡 盲盒揭晓</div>" + instanceData[animState.currentModel][testModeMapping[name]];
                        tooltipPos = "bottom"; borderColor = "#e879f9";
                        // 揭晓时稍微放大对应区块
                        if(name.includes('pos_') || name === 'oxidant') animState.greenCubeR = 1.05;
                        if(name.includes('neg_') || name === 'reductant') animState.greenCubeL = 1.05;
                    } 
                    // 【常态教学互动】
                    else {
                        if (name === 'ion_migration_label') { tooltipText = "阳离子(红)向正极，阴离子(紫)向负极"; tooltipPos = "left"; borderColor = "#f87171"; forceHighlightIon = true; } 
                        else if (name === 'ion_conductor_text') { tooltipText = (animState.currentModel === 'abstract') ? "<div class='tooltip-title'>离子导体实例</div>溶液、盐桥、离子交换膜 等" : "承载离子迁移的内部通道"; tooltipPos = "bottom"; borderColor = "#ef4444"; animState.redCube = 1.05; forceHighlightIon = true; } 
                        else if (name === 'electron_conductor_text') { tooltipText = (animState.currentModel === 'abstract') ? "<div class='tooltip-title'>电子导体实例</div>电极材料、导线、用电器 等" : "引导电子流转的外部通道"; tooltipPos = "bottom"; borderColor = "#60a5fa"; animState.blueCube = 1.05; forceHighlightElectron = true; } 
                        else if (name === 'electron_transfer_label') { tooltipText = "电子由负极流出，经外电路注入正极"; tooltipPos = "left"; borderColor = "#60a5fa"; forceHighlightElectron = true; } 
                        else if (name === 'pos_material_text') { if (animState.currentModel === 'abstract') tooltipText = "<div class='tooltip-title'>正极材料实例</div>不活泼金属、石墨棒等"; tooltipPos = "bottom"; borderColor = "#6ee7b7"; animState.greenCubeR = 1.05; } 
                        else if (name === 'neg_material_text') { if (animState.currentModel === 'abstract') tooltipText = "<div class='tooltip-title'>负极材料实例</div>活泼金属材料、导电非金属等"; tooltipPos = "bottom"; borderColor = "#6ee7b7"; animState.greenCubeL = 1.05; } 
                        else if (name === 'g1_label' || name === 'electrode_reactants') { forceG1 = true; } else if (name === 'g2_label' || name === 'electrode_products') { forceG2 = true; }
                    }
                } else { document.body.style.cursor = 'default'; }

                if (tooltipText) {
                    tooltip.innerHTML = tooltipText; tooltip.style.borderLeftColor = borderColor; tooltip.style.display = 'block';
                    const offX = 25; const offY = -40; 
                    if (tooltipPos === "right") { tooltip.style.left = (mouseX + offX) + 'px'; tooltip.style.top = (mouseY + offY) + 'px'; tooltip.style.transform = 'translate(0, -50%)'; } 
                    else if (tooltipPos === "left") { tooltip.style.left = (mouseX - offX) + 'px'; tooltip.style.top = (mouseY + offY) + 'px'; tooltip.style.transform = 'translate(-100%, -50%)'; } 
                    else if (tooltipPos === "bottom") { tooltip.style.left = mouseX + 'px'; tooltip.style.top = (mouseY + offY) + 'px'; tooltip.style.transform = 'translate(-50%, -100%)'; }
                } else { tooltip.style.display = 'none'; }
            }

            const isActive = animState.isCircuitClosed || animState.isTouring;
            const glowG1 = isActive || forceG1;
            const glowG2 = isActive || forceG2;

            applyGlowAnimation(parts.ox_product, glowG2, time); applyGlowAnimation(parts.red_product, glowG2, time); applyGlowAnimation(parts.eq_ox_product, glowG2, time); applyGlowAnimation(parts.eq_red_product, glowG2, time); applyGlowAnimation(parts.eq_g_diff, glowG2, time);     

            // 离子运动逻辑

                } else { e.mesh.visible = false; }
            });

            controls.update(); composer.render();
        }
