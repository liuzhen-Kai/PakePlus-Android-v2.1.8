        function createPBRAtom(r, pbrMat) { return new THREE.Mesh(new THREE.SphereGeometry(r, 32, 32), pbrMat.clone()); }
        function createPBRBond(p1, p2) {
            const d = p1.distanceTo(p2);
            const cyl = new THREE.Mesh(new THREE.CylinderGeometry(0.08, 0.08, d, 8), PBR_MATS.Bond.clone());
            cyl.position.copy(p1).lerp(p2, 0.5); cyl.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), p2.clone().sub(p1).normalize());
            return cyl;

        function buildPBRCrystal(type) {
            const group = new THREE.Group();
            if (type === 'MnO2' || type === 'MnOOH') {
                const spacing = 0.7; 
                for(let x=-1; x<=1; x++) {
                    for(let y=-1; y<=1; y++) {
                        for(let z=-1; z<=1; z++) {
                            const isMn = (Math.abs(x) + Math.abs(y) + Math.abs(z)) % 2 === 0;
                            if (isMn) {
                                const mn = createPBRAtom(0.45, PBR_MATS.Mn); mn.position.set(x*spacing, y*spacing, z*spacing); group.add(mn);
                            } else {
                                const o = createPBRAtom(0.35, PBR_MATS.O); o.position.set(x*spacing, y*spacing, z*spacing); group.add(o);
                                if (type === 'MnOOH' && Math.random() > 0.4) {
                                    const h = createPBRAtom(0.2, PBR_MATS.H);
                                    h.position.set(x*spacing + Math.sign(x)*0.25, y*spacing + Math.sign(y)*0.25, z*spacing + Math.sign(z)*0.25);
                                    if(x===0 && y===0 && z===0) h.position.set(0.25, 0.25, 0.25);
                                    group.add(h);
                                }
                            }
                        }
                    }
                }
            } else if (type === 'Zn') {
                const spacing = 0.8;
                for(let x=-1; x<=1; x++) {
                    for(let y=-1; y<=1; y++) {
                        for(let z=-1; z<=1; z++) {
                            const zn = createPBRAtom(0.45, PBR_MATS.Zn); zn.position.set(x*spacing, y*spacing, z*spacing); group.add(zn);
                        }
                    }
                }
            }
            return group;
        }

            }

            const targetPos = isDriving ? posCar : posInspect;
            const targetRot = isDriving ? rotCar : rotInspect;
            batteryGroup.position.lerp(targetPos, 0.1);
            batteryGroup.quaternion.slerpQuaternions(batteryGroup.quaternion, new THREE.Quaternion().setFromEuler(targetRot), 0.1);

            if (isDriving && batteryPower > 0) {
                batteryPower -= (isTouring ? 0.18 : 0.12); 
                if (batteryPower <= 0) {
                    batteryPower = 0;
                    document.getElementById('btn-replace').style.display = 'block';
                    document.getElementById('btn-action').style.display = 'none';
                }
            }
            
            document.getElementById('power-fill').style.transform = `scaleX(${batteryPower / 100})`;
            const pt = document.getElementById('power-percent');
            pt.innerText = Math.ceil(batteryPower) + '%';
            pt.style.color = batteryPower < 20 ? '#ff0055' : (batteryPower < 50 ? '#f59e0b' : '#10b981');

            const targetSpeed = (isDriving && batteryPower > 0) ? (batteryPower/100 * 0.4 + 0.1) : 0;
            wheelSpeed = THREE.MathUtils.lerp(wheelSpeed, targetSpeed, 0.02);
            if (wheelSpeed > 0.001) {
                wheels.forEach(w => w.rotation.z -= wheelSpeed);
                carGroup.position.y = 0 + Math.random() * (wheelSpeed * 0.1); 
            } else { carGroup.position.y = 0; }

            circuitElectrons.forEach((el, i) => {
                if (isDriving && batteryPower > 0) {
                    el.visible = true;
                    const flowSpeed = (batteryPower / 100) * 1.5 + 0.5;
                    const t = (time * flowSpeed * 0.2 + i / numElectrons) % 1;
                    const pt = circuitCurve.getPointAt(t);
                    el.position.copy(pt);
                } else {
                    el.visible = false;
                }
            });

            // 核心修复：在这里同步微观粒子的 Y 轴偏移
            if (viewMode === 'micro' && mechMode === null) {
                const powerRatio = batteryPower / 100;
                particles.forEach((cloud, idx) => {
                    const data = cloud.userData;
                    const explodeY = macroParts[layerMap[idx]].parentGroup.position.y; // 完美获取对应的 Y 轴偏移量
                    
                    for(let i=0; i<data.pData.length; i++) {
                        const p = data.pData[i];
                        const activeScale = data.isProduct ? p.baseScale * Math.max((1 - powerRatio), 0.01) : p.baseScale * Math.max(powerRatio, 0.01);
                        const wY = p.y + explodeY + Math.sin(time*3+i)*0.02; // 添加偏移
                        dummyAnim.position.set(p.r*Math.cos(p.a), wY, p.r*Math.sin(p.a));
                        dummyAnim.rotation.set(p.rotX+time*0.5, p.rotY, p.rotZ);
                        dummyAnim.scale.setScalar(activeScale);
                        dummyAnim.updateMatrix(); cloud.setMatrixAt(i, dummyAnim.matrix);
                    }
                    cloud.instanceMatrix.needsUpdate = true;
                });
            }

            updateLabelsPos();
            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        animate();
    </script>
