        function createBond(p1, p2) {
            const geo = new THREE.CylinderGeometry(0.04, 0.04, p1.distanceTo(p2), 8);
            const bond = new THREE.Mesh(geo, new THREE.MeshStandardMaterial({color: 0x888888, roughness: 0.5}));
            bond.position.copy(p1.clone().lerp(p2, 0.5));
            bond.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), p2.clone().sub(p1).normalize());
            return bond;
        }

        function createMolecule(type) {
            const group = new THREE.Group();
            if (type === 'H2') {
                const s1 = new THREE.Mesh(new THREE.SphereGeometry(0.18, 16, 16), new THREE.MeshStandardMaterial({color: 0xffffff, roughness: 0.2}));
                const s2 = s1.clone(); s1.position.x = -0.16; s2.position.x = 0.16;
                group.add(s1, s2, createBond(s1.position, s2.position));
            } else if (type === 'O2') {
                const s1 = new THREE.Mesh(new THREE.SphereGeometry(0.24, 16, 16), new THREE.MeshStandardMaterial({color: 0xef4444, roughness: 0.3}));
                const s2 = s1.clone(); s1.position.x = -0.2; s2.position.x = 0.2;
                const bond = createBond(s1.position, s2.position); bond.scale.set(1.5, 1, 1.5);
                group.add(s1, s2, bond);
            } else if (type === 'H2O') {
                const o = new THREE.Mesh(new THREE.SphereGeometry(0.24, 16, 16), new THREE.MeshStandardMaterial({color: 0xef4444, roughness: 0.3}));
                const h1 = new THREE.Mesh(new THREE.SphereGeometry(0.15, 16, 16), new THREE.MeshStandardMaterial({color: 0xffffff, roughness: 0.2}));
                const h2 = h1.clone(); o.position.set(0, 0.1, 0); h1.position.set(-0.22, -0.12, 0); h2.position.set(0.22, -0.12, 0);
                group.add(o, h1, h2, createBond(o.position, h1.position), createBond(o.position, h2.position));
            } else if (type === 'H+') {
                group.add(new THREE.Mesh(new THREE.SphereGeometry(0.14, 16, 16), new THREE.MeshStandardMaterial({color: 0x00ffff, emissive: 0x00ffff, emissiveIntensity: 2.5, roughness: 0.1})));
            } else if (type === 'e-') {
                group.add(new THREE.Mesh(new THREE.SphereGeometry(0.12, 8, 8), new THREE.MeshStandardMaterial({color: 0xfde047, emissive: 0xfde047, emissiveIntensity: 4.5})));
            }
            return group;
        }

        function createBipolarTexture() {
            const canvas = document.createElement('canvas'); canvas.width = 512; canvas.height = 512; const ctx = canvas.getContext('2d');
            ctx.fillStyle = '#a0a5a9'; ctx.fillRect(0, 0, 512, 512); ctx.strokeStyle = '#6c7278'; ctx.lineWidth = 18; ctx.lineCap = 'round'; ctx.lineJoin = 'round';
            ctx.beginPath(); let y = 40, dir = 1; ctx.moveTo(40, y);
            while (y < 480) {
                if (dir===1) { ctx.lineTo(470, y); ctx.arc(470, y+20, 20, 1.5*Math.PI, 0.5*Math.PI, false); y+=40; ctx.moveTo(470, y); } 
                else { ctx.lineTo(40, y); ctx.arc(40, y+20, 20, 1.5*Math.PI, 0.5*Math.PI, true); y+=40; ctx.moveTo(40, y); } dir *= -1;
            } ctx.stroke(); return new THREE.CanvasTexture(canvas);
        }

        function createMEATexture() {
            const canvas = document.createElement('canvas'); canvas.width = 512; canvas.height = 512; const ctx = canvas.getContext('2d');
            ctx.fillStyle = '#87ceeb'; ctx.fillRect(0, 0, 512, 512); ctx.strokeStyle = '#1a56db'; ctx.lineWidth = 60; ctx.strokeRect(30, 30, 452, 452);
            return new THREE.CanvasTexture(canvas);
        }

        function createPipe(parentGroup, x, y, length, type, isInput) {
            const pipe = new THREE.Mesh(new THREE.CylinderGeometry(0.6, 0.6, Math.abs(length), 16), new THREE.MeshStandardMaterial({ color: 0x888888, transparent:true, opacity:0.5 }));
            pipe.rotation.z = Math.PI / 2; pipe.position.set(x + length/2, y, 0); parentGroup.add(pipe);
            let lText = '', lClass = '', mType = '';
            if (type==='H2_in') {lText='H₂ 进气'; lClass='red'; mType='H2';} else if (type==='H2_out') {lText='未反应 H₂'; lClass='gray'; mType='H2';}
            else if (type==='O2_in') {lText='O₂ 进气'; lClass='green'; mType='O2';} else if (type==='H2O_out') {lText='H₂O 产水'; lClass='blue'; mType='H2O';}
            parentGroup.add(createLabel(lText, lClass, new THREE.Vector3(x+length+(length>0?1.5:-1.5), y+1.5, 0), false));
            const sX = isInput ? (x+length) : x, eX = isInput ? x : (x+length), mDir = isInput ? (length>0?-1:1) : (length>0?1:-1);
            for (let i=0; i<4; i++) {
                const mol = createMolecule(mType); mol.position.set(x + Math.random()*length, y+(Math.random()-0.5)*0.5, (Math.random()-0.5)*0.5);
                parentGroup.add(mol); pipeMolecules.push({ mesh: mol, startX: sX, endX: eX, dirX: mDir, baseSpeed: 0.02 + Math.random()*0.02 });
            }
        }

        casing.rotation.x = Math.PI / 2;

        function updateWires(spacing) {
            const anodeZ = parts[1].userData.baseZ + parts[1].userData.explodeDir * spacing;
            const cathodeZ = parts[3].userData.baseZ + parts[3].userData.explodeDir * spacing;
            const plateTopY = plateHeight / 2;

        }

        microGroup.add(lblPt);

        const nafionBlob = new THREE.Mesh(new THREE.SphereGeometry(3, 32, 32), new THREE.MeshStandardMaterial({color: 0x3b82f6, transparent: true, opacity: 0.4, roughness: 0.1, depthWrite: false}));
        nafionBlob.scale.set(1, 1.2, 0.5); nafionBlob.position.set(0, 0, 0.5); microGroup.add(nafionBlob); 
        lblNafion = createLabel('③ 质子交换膜 (吸收H⁺)', 'blue', new THREE.Vector3(6.5, 3, 1.0), true);
        microGroup.add(lblNafion);

        const microH2 = createMolecule('H2'); 
        const microH_ads1 = new THREE.Mesh(new THREE.SphereGeometry(0.35, 16, 16), new THREE.MeshStandardMaterial({color: 0xffffff, roughness: 0.2})); const microH_ads2 = microH_ads1.clone();
        const microHPlus1 = new THREE.Mesh(new THREE.SphereGeometry(0.3, 16, 16), new THREE.MeshStandardMaterial({color: 0x00ffff, emissive: 0x00ffff, emissiveIntensity: 2.5})); const microHPlus2 = microHPlus1.clone();
        const microE1 = new THREE.Mesh(new THREE.SphereGeometry(0.2, 16, 16), new THREE.MeshStandardMaterial({color: 0xfde047, emissive: 0xfde047, emissiveIntensity: 3.5})); const microE2 = microE1.clone();
        microGroup.add(microH2, microH_ads1, microH_ads2, microHPlus1, microHPlus2, microE1, microE2);

        // ==========================================
        // 6. 宏观电池粒子流

        const migratingParticles = [];
        function addMigrationStream(type, startIdx, endIdx, count) {
            for(let i=0; i<count; i++) {
                const mol = createMolecule(type); scene.add(mol); 
                migratingParticles.push({ mesh: mol, type: type, startIdx: startIdx, endIdx: endIdx, progress: Math.random(), baseSpeed: 0.0015+Math.random()*0.002, offsetX: (Math.random()-0.5)*8, offsetY: (Math.random()-0.5)*12 });
            }
        }
        addMigrationStream('H2', 0, 1, 20); addMigrationStream('O2', 4, 3, 20); addMigrationStream('H2O', 3, 4, 25);
        for(let i=0; i<45; i++) { const h = createMolecule('H+'); scene.add(h); migratingParticles.push({ mesh: h, type: 'H+跨膜', progress: Math.random(), baseSpeed: 0.0025+Math.random()*0.0025, offsetX: (Math.random()-0.5)*8, offsetY: (Math.random()-0.5)*12 }); }
        for(let i=0; i<25; i++) { const e = createMolecule('e-'); scene.add(e); migratingParticles.push({ mesh: e, type: '电子', progress: Math.random(), baseSpeed: 0.003+Math.random()*0.002 }); }

        // ==========================================
        // 8. 核心渲染循环
        // ==========================================
        function animate() {
            requestAnimationFrame(animate);
            const delta = clock.getDelta();

            // 目标电流随开关而定
            let targetCurrent = isAnimating ? (speedMultiplier * baseCurrent) : 0;
            let actualCurrent = targetCurrent > 0 ? targetCurrent + (Math.random()*0.1 - 0.05) : 0;
            
            let actualVoltage = actualCurrent > 0 ? (1.23 - 0.08 * actualCurrent + (Math.random()*0.005)) : 0;
            if(actualVoltage < 0) actualVoltage = 0;
            let actualPower = actualVoltage * actualCurrent; 
            let targetRpm = actualPower * 400; 

            DOM.valCurr.textContent = actualCurrent.toFixed(2) + ' A'; DOM.valVolt.textContent = actualVoltage.toFixed(2) + ' V';
            DOM.valPower.textContent = actualPower.toFixed(2) + ' W'; DOM.valRpm.textContent = Math.floor(targetRpm) + ' RPM';
            bladeMat.emissiveIntensity = actualPower > 0 ? (1.0 + actualPower * 0.3) : 0; 

            if (isCameraMoving) {
                camera.position.lerp(targetCameraPos, 0.05); controls.target.lerp(targetControlsCenter, 0.08);
                if (camera.position.distanceTo(targetCameraPos) < 0.2) isCameraMoving = false; 
            }

            fanGroup.rotation.z += (targetRpm * 0.0005) + (actualPower > 0 ? 0.02 : 0); 

            if (isAnimating) {
                pipeMolecules.forEach(p => {
                    p.mesh.position.x += p.dirX * p.baseSpeed * speedMultiplier; p.mesh.rotation.x += 0.05 * speedMultiplier; p.mesh.rotation.y += 0.05 * speedMultiplier;
                    if (p.dirX > 0 && p.mesh.position.x > Math.max(p.startX, p.endX)) { p.mesh.position.x = Math.min(p.startX, p.endX); } else if (p.dirX < 0 && p.mesh.position.x < Math.min(p.startX, p.endX)) { p.mesh.position.x = Math.max(p.startX, p.endX); }
                });

                migratingParticles.forEach(p => {
                    p.progress += p.baseSpeed * speedMultiplier;
                    if (p.progress > 1.0) { p.progress = 0; p.offsetX = (Math.random()-0.5)*8; p.offsetY = (Math.random()-0.5)*12; }
                    
                    let tz=0, tx=p.offsetX, ty=p.offsetY;
                    
                    if (p.type === 'H+跨膜') {
                        const z1 = parts[1].position.z, z2 = parts[2].position.z, z3 = parts[3].position.z;
                        if (p.progress < 0.5) tz = z1 + (z2 - z1) * (p.progress*2); else tz = z2 + (z3 - z2) * ((p.progress-0.5)*2);
                    } else if (p.type === '电子') {
                        if (p.progress < 0.45) {
                            const t = p.progress / 0.45; const pos = curveWire1.getPoint(t); tx = pos.x; ty = pos.y; tz = pos.z;
                        } else if (p.progress < 0.55) {
                            const t = (p.progress - 0.45) / 0.1; tx = termLeft.x + (termRight.x - termLeft.x) * t; ty = motorY; tz = 0;
                        } else {
                            const t = (p.progress - 0.55) / 0.45; const pos = curveWire2.getPoint(t); tx = pos.x; ty = pos.y; tz = pos.z;
                        }
                        tx += (Math.random()-0.5)*0.1; ty += (Math.random()-0.5)*0.1; tz += (Math.random()-0.5)*0.1;
                    } else { 
                        tz = parts[p.startIdx].position.z + (parts[p.endIdx].position.z - parts[p.startIdx].position.z) * p.progress; 
                    }

                    if (p.type !== '电子') { 
                        p.offsetX += (Math.random()-0.5)*0.06*speedMultiplier; p.offsetY += (Math.random()-0.5)*0.06*speedMultiplier;
                        if (p.offsetX > 4.5) p.offsetX = 4.5; if (p.offsetX < -4.5) p.offsetX = -4.5; if (p.offsetY > 6.5) p.offsetY = 6.5; if (p.offsetY < -6.5) p.offsetY = -6.5;
                        tx = p.offsetX; ty = p.offsetY;
                    } 
                    p.mesh.position.set(tx, ty, tz);
                    if (p.type !== 'H+跨膜' && p.type !== '电子') { p.mesh.rotation.x += 0.05*speedMultiplier; p.mesh.rotation.y += 0.05*speedMultiplier; }
                });
            }

            // ⭐ 幻灯片式分布高亮解说 (微观视角)
            if (inMicroView) {
                if(isAnimating) {
                    microTime += delta * speedMultiplier * 0.8; 
                    const cycle = microTime % 4.0; 
                    const ptTargetZ = -0.8; 
                    
                    reactionLabel.element.style.opacity = "1";

                    if (cycle < 1.0) {
                        reactionLabel.element.innerHTML = '步骤 1: H₂ 抵达负极<br><span style="color:#fbbf24">H₂(g) → 穿透碳纸进入催化层</span>';
                        lblCarbon.element.style.opacity = "1"; lblPt.element.style.opacity = "0.2"; lblNafion.element.style.opacity = "0.2";
                        
                        microH2.visible = true; microH_ads1.visible = false; microH_ads2.visible = false;
                        microHPlus1.visible = false; microHPlus2.visible = false; microE1.visible = false; microE2.visible = false;
                        microH2.position.set(0, 0, -6 + cycle * 5.2); microH2.rotation.z = cycle * Math.PI; 
                    } else if (cycle < 2.0) {
                        reactionLabel.element.innerHTML = '步骤 2: 负极解离吸附<br><span style="color:#fbbf24">H₂ + 2Pt ⇌ 2Pt-H</span>';
                        lblCarbon.element.style.opacity = "0.2"; lblPt.element.style.opacity = "1"; lblNafion.element.style.opacity = "0.2";
                        
                        microH2.visible = false; microH_ads1.visible = true; microH_ads2.visible = true; 
                        const t = cycle - 1.0; 
                        microH_ads1.position.set(-0.4 - t*0.5, 0.2 + t*0.2, ptTargetZ); microH_ads2.position.set(0.4 + t*0.5, -0.2 - t*0.2, ptTargetZ);
                    } else if (cycle < 2.5) {
                        reactionLabel.element.innerHTML = '步骤 3: 负极氧化反应 (核心)<br><span style="color:#ef4444">Pt-H ⇌ Pt + H⁺ + e⁻ (失去电子)</span>';
                        lblCarbon.element.style.opacity = "0.2"; lblPt.element.style.opacity = "1"; lblNafion.element.style.opacity = "0.2";

                        microH_ads1.visible = false; microH_ads2.visible = false;
                        microHPlus1.visible = true; microHPlus2.visible = true; microE1.visible = true; microE2.visible = true;
                        
                        microHPlus1.position.set(-0.9, 0.4, ptTargetZ); microHPlus2.position.set(0.9, -0.4, ptTargetZ);
                        microE1.position.set(-0.9, 0.4, ptTargetZ - 0.2); microE2.position.set(0.9, -0.4, ptTargetZ - 0.2);
                    } else {
                        reactionLabel.element.innerHTML = '步骤 4: 粒子分流<br><span style="color:#fbbf24">e⁻ 退回扩散层 / H⁺ 跨入交换膜</span>';
                        lblCarbon.element.style.opacity = "1"; lblPt.element.style.opacity = "0.2"; lblNafion.element.style.opacity = "1";

                        const flyT = cycle - 2.5; 
                        microE1.position.z = (ptTargetZ - 0.2) - flyT * 1; microE1.position.y = 0.4 + flyT * 4;
                        microE2.position.z = (ptTargetZ - 0.2) - flyT * 1; microE2.position.y = -0.4 + flyT * 4;
                        microHPlus1.position.z = ptTargetZ + flyT * 3; microHPlus2.position.z = ptTargetZ + flyT * 3;
                    }
                } else {
                    reactionLabel.element.innerHTML = '系统未运行<br><span style="color:#94a3b8">点击下方“开始工作”触发反应</span>';
                    lblCarbon.element.style.opacity = "0.2"; lblPt.element.style.opacity = "0.2"; lblNafion.element.style.opacity = "0.2";
                }
            }

            controls.update(); composer.render(); labelRenderer.render(scene, camera);
        }

        animate();
    </script>
