        function playSound(type, freq, duration, vol, sweepTo) {
            if(audioCtx.state === 'suspended') audioCtx.resume();
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = type;
            osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
            if(sweepTo) osc.frequency.exponentialRampToValueAtTime(sweepTo, audioCtx.currentTime + duration);
            gain.gain.setValueAtTime(vol, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);
            osc.connect(gain); gain.connect(audioCtx.destination);
            osc.start(); osc.stop(audioCtx.currentTime + duration);
        }

                            triggerManualO2Fusion(draggedObj, ion.mesh); break;
                        }
                    }
                } else if (name.includes("Al³⁺")) {
                    if(draggedObj.position.y < eYMin + 1.2) triggerManualAlDeposition(draggedObj);
                }
                draggedObj = null; controls.enabled = true;
            }
        });

        function triggerManualO2Fusion(m1, m2) {
            const ion1 = oIons.find(i => i.mesh === m1); const ion2 = oIons.find(i => i.mesh === m2);
            ion1.state = 'binding'; ion2.state = 'binding';
            const mid = m1.position.clone().add(m2.position).multiplyScalar(0.5);
            m1.position.set(mid.x - 0.15, mid.y, mid.z); m2.position.set(mid.x + 0.15, mid.y, mid.z);
            for(let j=0; j<4; j++) {
                let el = createReactionElectron(); el.position.copy(mid); el.userData = { target: null, progress: 0, dir: new THREE.Vector3((Math.random()-0.5)*0.4, 1, (Math.random()-0.5)*0.4) }; reactionElectrons.push(el); particlesGroup.add(el);
            }
            let o2 = o2Pool.find(m => !m.active); if(o2) { o2.active = true; o2.mesh.visible = true; o2.mesh.position.copy(mid); }
            
            playSpark();
            if(questActive) { questO2++; document.getElementById('quest-o2').innerText = `生成氧气: ${questO2} / 2`; checkQuest(); }

            setTimeout(() => {
                ion1.state = 'rising'; ion2.state = 'rising';
                ion1.mesh.position.set(THREE.MathUtils.randFloat(eXMin, eXMax), eYMin, THREE.MathUtils.randFloat(eZMin, eZMax));
                ion2.mesh.position.set(THREE.MathUtils.randFloat(eXMin, eXMax), eYMin, THREE.MathUtils.randFloat(eZMin, eZMax));
            }, 500);
        }

        function triggerManualAlDeposition(m) {
            const ion = alIons.find(i => i.mesh === m);
            if(ion) {
                ion.state = 'receiving_electrons'; ion.timer = 0;
                for(let j=0; j<3; j++) {
                    const el = createReactionElectron(); el.position.set(ion.mesh.position.x + (Math.random()-0.5)*1.5, eYMin - 0.2, ion.mesh.position.z + (Math.random()-0.5)*1.5);
                    el.userData = { target: ion.mesh, progress: 0, speed: 0.01 + Math.random()*0.005 }; reactionElectrons.push(el); particlesGroup.add(el);
                }
                playDrop();
                if(questActive) { questAl++; document.getElementById('quest-al').innerText = `提取液态铝: ${questAl} / 3`; checkQuest(); }
            }
        }

        function gsapCameraTo(x, y, z, tx, ty, tz) {
            const startPos = camera.position.clone(), endPos = new THREE.Vector3(x, y, z);
            const startTarget = controls.target.clone(), endTarget = new THREE.Vector3(tx, ty, tz);
            let p = 0; const anim = () => { p += 0.03; if(p <= 1) { const e = 1 - Math.pow(1 - p, 3); camera.position.lerpVectors(startPos, endPos, e); controls.target.lerpVectors(startTarget, endTarget, e); requestAnimationFrame(anim); } }; anim();
        }

        function animate() {

        function render() {
            controls.update();

            // --- 危机模拟：消耗与阳极效应 ---
            if (window.isPlaying && window.viewMode !== 'interactive') {
                aluminaConc = Math.max(0.0, aluminaConc - 0.0005);
                document.getElementById('conc-bar').style.width = (aluminaConc * 100) + '%';
                
                if (aluminaConc < 0.15 && !isAnodeEffect) {
                    isAnodeEffect = true;
                    document.getElementById('alarm-overlay').classList.remove('hidden');
                    document.getElementById('voltage-txt').innerText = "38.5 V 飙升!";
                    document.getElementById('voltage-txt').className = "text-red-600 font-bold animate-pulse";
                    btnFeed.classList.remove('hidden');
                    if(!alarmInterval) alarmInterval = setInterval(playAlarmBeep, 600);
                }
            }

            if (window.hasPower && window.powerT < 1.0) window.powerT += 0.01;
            if (!window.hasPower && window.powerT > 0.0) window.powerT -= 0.01;

            const curColor = isAnodeEffect ? errorColor : hotColor;
            const curEmissive = isAnodeEffect ? emissiveError : emissiveHot;

            heatLight.intensity = (isAnodeEffect ? 3.0 : 1.2) * window.powerT;
            heatLight.color.lerpColors(new THREE.Color(0xff6600), errorColor, isAnodeEffect ? 1 : 0);
            materials.electrolyteMacro.color.lerpColors(coldColor, curColor, window.powerT);
            materials.electrolyteMacro.emissive.lerpColors(emissiveCold, curEmissive, window.powerT);

            const time = Date.now() * 0.004;
            for(let i=0; i<alPosAttr.count; i++) {
                if (alPoolGeom.userData.baseY[i] > 0.5) { 
                    const x = alPosAttr.getX(i); const z = alPosAttr.getZ(i);
                    const wave = (Math.sin(x * 1.5 + time) + Math.cos(z * 1.5 + time * 0.8)) * (isAnodeEffect ? 0.15 : 0.08) * window.powerT;
                    alPosAttr.setY(i, alPoolGeom.userData.baseY[i] + wave);
                }
            }
            alPosAttr.needsUpdate = true;

            // 射线提示
            raycaster.setFromCamera(mouse, camera);
            const intersects = raycaster.intersectObjects(scene.children, true);
            let foundName = null; let macroName = null;
            for (let i = 0; i < intersects.length; i++) {
                let obj = intersects[i].object; let isVisible = true; let curr = obj;
                while (curr) { if (!curr.visible) { isVisible = false; break; } curr = curr.parent; }
                if (!isVisible) continue;
                let objName = obj.userData.name || (obj.parent && obj.parent.userData.name);
                if (objName) {
                    if (objName.includes("Al") || objName.includes("O²") || objName.includes("e⁻") || objName.includes("O₂") || objName.includes("Na")) {
                        if (window.viewMode !== 'macro') { foundName = objName; break; }
                    } else if (!macroName) { macroName = objName; }
                }
            }
            foundName = foundName || macroName;
            if(window.powerT < 0.2 && foundName && window.viewMode !== 'macro') foundName += " (冻结固化)";
            if (foundName && !draggedObj) { tooltip.innerText = foundName; tooltip.classList.remove('hidden'); tooltip.classList.remove('opacity-0'); } 
            else { tooltip.classList.add('opacity-0'); tooltip.classList.add('hidden'); }

            const autoMove = window.isPlaying && window.viewMode !== 'interactive' && !isAnodeEffect;

            if (window.viewMode !== 'macro') {
                alIons.forEach(ion => {
                    if (ion.state === 'falling') {
                        if (autoMove && ion.mesh !== draggedObj) {
                            ion.mesh.position.y -= ion.speed * window.powerT;
                            if (ion.mesh.position.y <= eYMin + 0.8) {
                                ion.state = 'receiving_electrons'; ion.timer = 0;
                                for(let j=0; j<3; j++) {
                                    const el = createReactionElectron(); el.position.set(ion.mesh.position.x + (Math.random()-0.5)*1.5, eYMin - 0.2, ion.mesh.position.z + (Math.random()-0.5)*1.5);
                                    el.userData = { target: ion.mesh, progress: 0, speed: 0.01 + Math.random()*0.005 }; reactionElectrons.push(el); particlesGroup.add(el);
                                }
                            }
                        }
                    } else if (ion.state === 'receiving_electrons') {
                        ion.timer += 0.5 * window.powerT;
                        if (ion.timer > 40) { ion.state = 'deposited'; ion.timer = 0; ion.mesh.material = matAlMetal; ion.mesh.userData.name = "Al 铝单质"; }
                    } else if (ion.state === 'deposited') {
                        ion.timer += 0.5 * window.powerT; ion.mesh.position.y -= 0.0075 * window.powerT; ion.mesh.scale.addScalar(-0.01 * window.powerT);
                        if (ion.timer > 40) { ion.state = 'falling'; ion.mesh.material = matAl; ion.mesh.userData.name = "Al³⁺"; ion.mesh.scale.setScalar(1); ion.mesh.position.set(THREE.MathUtils.randFloat(eXMin, eXMax), eYMax, THREE.MathUtils.randFloat(eZMin, eZMax)); }
                    }
                });

                oIons.forEach(ion => {
                    if (ion.state === 'rising') {
                        if (autoMove && ion.mesh !== draggedObj) {
                            if (!ion.targetSite) { let avail = anodeSites.filter(s => s.count < 2); if(avail.length > 0) ion.targetSite = avail[Math.floor(Math.random() * avail.length)]; }
                            if (ion.targetSite) {
                                let dx = ion.targetSite.x - ion.mesh.position.x, dy = ion.targetSite.y - ion.mesh.position.y, dz = ion.targetSite.z - ion.mesh.position.z;
                                let dist = Math.sqrt(dx*dx + dy*dy + dz*dz);
                                if (dist < 0.1) {
                                    ion.state = 'binding'; ion.targetSite.count++; ion.targetSite.oxIons.push(ion);
                                    if (ion.targetSite.count === 1) { ion.mesh.position.set(ion.targetSite.x - 0.15, ion.targetSite.y, ion.targetSite.z); } 
                                    else if (ion.targetSite.count === 2) {
                                        let cSite = ion.targetSite; ion.mesh.position.set(cSite.x + 0.15, cSite.y, cSite.z);
                                        for(let j=0; j<4; j++) { let el = createReactionElectron(); el.position.copy(cSite); el.userData = { target: null, progress: 0, dir: new THREE.Vector3((Math.random()-0.5)*0.4, 1, (Math.random()-0.5)*0.4) }; reactionElectrons.push(el); particlesGroup.add(el); }
                                        let o2 = o2Pool.find(m => !m.active); if(o2) { o2.active = true; o2.mesh.visible = true; o2.mesh.position.copy(cSite); }
                                        cSite.oxIons.forEach(o => { o.state = 'rising'; o.targetSite = null; o.mesh.position.set(THREE.MathUtils.randFloat(eXMin, eXMax), eYMin, THREE.MathUtils.randFloat(eZMin, eZMax)); });
                                        cSite.count = 0; cSite.oxIons = [];
                                    }
                                } else { ion.mesh.position.add(new THREE.Vector3(dx/dist, dy/dist, dz/dist).multiplyScalar(ion.speed * window.powerT)); }
                            } else { ion.mesh.position.y += ion.speed * window.powerT; if(ion.mesh.position.y > eYMax) ion.mesh.position.y = eYMin; }
                        }
                    }
                });

                o2Pool.forEach(mol => {
                    if(mol.active) {
                        mol.mesh.position.y += mol.speed * window.powerT; mol.mesh.position.x += Math.sin(Date.now() * 0.001 + mol.mesh.position.y * 3) * 0.0025 * window.powerT;
                        mol.mesh.rotation.z += 0.02 * window.powerT; mol.mesh.rotation.y += 0.02 * window.powerT;
                        if (mol.mesh.position.y > eYMax + 2.0) { mol.active = false; mol.mesh.visible = false; }
                    }
                });

                for (let i = reactionElectrons.length - 1; i >= 0; i--) {
                    const el = reactionElectrons[i];
                    if (el.userData.target) {
                        el.userData.progress += el.userData.speed * window.powerT; el.position.lerpVectors(el.position, el.userData.target.position, 0.05 * window.powerT);
                    } else {
                        el.userData.progress += 0.01 * window.powerT; el.position.add(el.userData.dir.clone().multiplyScalar(0.04 * window.powerT));
                    }
                    if (el.userData.progress >= 1) { particlesGroup.remove(el); reactionElectrons.splice(i, 1); }
                }
            }

            if (window.hasPower && !isAnodeEffect) {
                externalElectrons.forEach(el => {
                    el.progress += 0.004 * window.powerT; if(el.progress > 1) el.progress = 0;
                    if(el.type === 'cathode') el.mesh.position.copy(blackCurve.getPoint(el.progress));
                    else el.mesh.position.copy(redCurve.getPoint(1 - el.progress));
                });
            }

            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        animate();
    </script>
