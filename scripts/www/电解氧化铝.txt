        function createBond(p1, p2, mat, radius=0.03) {
            const dist = p1.distanceTo(p2);
            const cyl = new THREE.Mesh(new THREE.CylinderGeometry(radius, radius, dist, 8), mat);
            cyl.position.copy(p1).lerp(p2, 0.5);
            cyl.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), p2.clone().sub(p1).normalize());
            return cyl;
        }

        function handleFusion(m1, m2) {
            playZap(); vibrate([30,30]); 
            const ion1 = oIons.find(i => i.mesh === m1); const ion2 = oIons.find(i => i.mesh === m2);
            ion1.state = 'binding'; ion2.state = 'binding';
            const mid = m1.position.clone().add(m2.position).multiplyScalar(0.5);
            m1.position.set(mid.x - 0.15, mid.y, mid.z); m2.position.set(mid.x + 0.15, mid.y, mid.z);
            for(let j=0; j<4; j++) { 
                let el = new THREE.Mesh(geomElectron, matElectron); el.position.copy(mid); el.userData = { target: null, progress: 0, dir: new THREE.Vector3((Math.random()-0.5), 1, (Math.random()-0.5)) }; reactionElectrons.push(el); particlesGroup.add(el);
            }
            let g = o2Pool.find(p=>!p.active); if(g){ g.active=true; g.mesh.visible=true; g.mesh.position.copy(mid); }
            if(mission.active && !mission.done){ mission.o2++; updateMissionUI(); }
            setTimeout(() => {
                ion1.state = 'rising'; ion2.state = 'rising';
                ion1.mesh.position.set(THREE.MathUtils.randFloat(eXMin, eXMax), eYMin, THREE.MathUtils.randFloat(eZMin, eZMax));
                ion2.mesh.position.set(THREE.MathUtils.randFloat(eXMin, eXMax), eYMin, THREE.MathUtils.randFloat(eZMin, eZMax));
            }, 500);
        }

        ui.btnFeed.onclick = ()=>{ initAudio(); vibrate(100); aluminaConc = 1.0; isAnodeEffect = false; document.getElementById('alarm-overlay').classList.add('opacity-0'); document.getElementById('voltage-label').className="font-mono font-bold text-green-600"; let ap=0; powderPool.forEach(p=>{if(!p.active&&ap<15){p.active=true;p.mesh.visible=true;p.mesh.position.set((Math.random()-0.5)*8,eYMax+3,(Math.random()-0.5)*4);p.vy=-0.1-Math.random()*0.05;p.vx=(Math.random()-0.5)*0.02;ap++;}}); };
        ui.btnCrystal.onclick = () => { prevViewMode = window.viewMode; window.viewMode = 'crystal'; initAudio(); vibrate(50); updateView(); };
        ui.btnCrystalClose.onclick = () => { window.viewMode = prevViewMode; initAudio(); vibrate(50); updateView(); };

        function gsapCameraTo(x,y,z){ const sp=camera.position.clone(),ep=new THREE.Vector3(x,y,z); let p=0; const a=()=>{p+=0.04;if(p<=1){camera.position.lerpVectors(sp,ep,1-Math.pow(1-p,3));requestAnimationFrame(a);}}; a(); }

        // --------------------------------------------------------
        // 7. 渲染循环 (含计时器、动画)
        // --------------------------------------------------------
        function animate() {
            window._aluminaAnimId = requestAnimationFrame(animate);
            let dt = clock.getDelta();
            
            controls.update();

            // 晶体加热熔化推演逻辑 (建议2)
            if(window.viewMode === 'crystal') {
                if(isCrystalHeating) {
                    al2o3Bonds.forEach(b => b.visible = false);
                    al2o3Nodes.forEach(n => {
                        if(!n.vel) n.vel = new THREE.Vector3(Math.random()-0.5, Math.random()-0.5, Math.random()-0.5).normalize().multiplyScalar(0.04);
                        n.mesh.position.add(n.vel);
                        if(Math.abs(n.mesh.position.x) > 2) n.vel.x *= -1;
                        if(Math.abs(n.mesh.position.y) > 2) n.vel.y *= -1;
                        if(Math.abs(n.mesh.position.z) > 2) n.vel.z *= -1;
                    });
                    
                    // 优化：氯化铝保持分子形态，在限制范围内做无规则热运动 (不直接飞出屏幕)
                    if (!al2cl6Group.userData.vel) al2cl6Group.userData.vel = new THREE.Vector3((Math.random()-0.5)*0.08, (Math.random()-0.5)*0.08, (Math.random()-0.5)*0.08);
                    al2cl6Group.position.add(al2cl6Group.userData.vel);
                    al2cl6Group.rotation.x += 0.02; al2cl6Group.rotation.z += 0.03;
                    
                    // 触碰边界反弹，始终保持在视口内易于观察
                    if(Math.abs(al2cl6Group.position.x - 3.5) > 1.5) al2cl6Group.userData.vel.x *= -1;
                    if(Math.abs(al2cl6Group.position.y - 3.0) > 1.5) al2cl6Group.userData.vel.y *= -1;
                    if(Math.abs(al2cl6Group.position.z) > 1.5) al2cl6Group.userData.vel.z *= -1;

                } else {
                    al2o3Group.rotation.y += 0.005; al2o3Group.rotation.x += 0.002;
                    al2cl6Group.rotation.y -= 0.01; al2cl6Group.rotation.z += 0.005;
                }
            }

            if (window.hasPower && window.powerT < 1.0) window.powerT += 0.02;
            if (!window.hasPower && window.powerT > 0.0) window.powerT -= 0.02;
            materials.electrolyteMacro.color.lerpColors(coldColor, hotColor, window.powerT);
            materials.electrolyteMacro.emissive.lerpColors(emissiveCold, emissiveHot, window.powerT);
            
            if (window.hasPower && window.viewMode !== 'crystal') {
                externalElectrons.forEach(el => {
                    el.progress += 0.005 * window.powerT; if(el.progress > 1) el.progress = 0;
                    if(el.type === 'cathode') el.mesh.position.copy(blackCurve.getPoint(el.progress));
                    else el.mesh.position.copy(redCurve.getPoint(1 - el.progress));
                });
            }

            if(window.hasPower) {
                if(isAnodeEffect) window.screenMat.color.setHex(Date.now()%400<200?0xff0000:0x330000);
                else window.screenMat.color.setRGB(0, (100+Math.sin(Date.now()*0.005)*50)/255, 0);
            } else window.screenMat.color.setHex(0x111111);

            if(window.isPlaying && window.viewMode !== 'crystal') {
                aluminaConc = Math.max(0, aluminaConc-0.0004);
                document.getElementById('conc-bar').style.width = (aluminaConc*100)+'%';
                if(aluminaConc > 0.3) document.getElementById('conc-bar').className = "bg-blue-500 h-full transition-all";
                else if(aluminaConc > 0.1) document.getElementById('conc-bar').className = "bg-yellow-500 h-full transition-all";
                else document.getElementById('conc-bar').className = "bg-red-600 h-full transition-all";

                if(aluminaConc < 0.1 && !isAnodeEffect) { isAnodeEffect=true; playAlarm(); document.getElementById('alarm-overlay').classList.remove('opacity-0'); document.getElementById('voltage-label').className="font-mono font-bold text-red-500 animate-pulse"; }
                
                // 任务计时器扣减 (建议3)
                if (mission.active && !mission.done && window.viewMode === 'interactive') {
                    let speed = isAnodeEffect ? 4.0 : 1.0; // 阳极效应加速掉时间！
                    missionTime -= dt * speed;
                    if(missionTime <= 0) {
                        missionTime = 0; mission.done = true; playAlarm();
                        showBanner("❌ 考核失败：时间耗尽！请注意打壳下料消除阳极效应！", false);
                    }
                    let tUI = document.getElementById('mission-timer');
                    if(tUI) {
                        tUI.innerText = `⏳ ${Math.ceil(missionTime)}s`;
                        if(missionTime < 15) tUI.classList.add('animate-pulse', 'text-red-600');
                    }
                }

                if(isAnodeEffect) {
                    document.getElementById('voltage-label').innerText = (35+Math.random()*5).toFixed(1)+" V";
                    heatLight.intensity = 3.0; heatLight.color.setHex(0xff0000);
                    if(Math.random()>0.8) vibrate(50);
                    if(Math.random()>0.5) { let s = new THREE.Mesh(geomElectron, new THREE.MeshBasicMaterial({color:0xffffff})); s.position.set([-3.5,0,3.5][Math.floor(Math.random()*3)]+(Math.random()-0.5), eYMax, (Math.random()-0.5)); s.userData = {vx:(Math.random()-0.5)*0.2,vy:0.1+Math.random()*0.2,vz:(Math.random()-0.5)*0.2,life:1.0}; scene.add(s); sparks.push(s); }
                } else {
                    document.getElementById('voltage-label').innerText = "4.2 V";
                    heatLight.intensity = 1.2 * window.powerT; heatLight.color.setHex(0xff6600);
                }
            } else {
                document.getElementById('voltage-label').innerText = "0.0 V";
            }

            for(let i=sparks.length-1; i>=0; i--) { let s=sparks[i]; s.position.x+=s.userData.vx; s.position.y+=s.userData.vy; s.position.z+=s.userData.vz; s.userData.vy-=0.01; s.userData.life-=0.05; s.scale.setScalar(Math.max(0,s.userData.life)); if(s.userData.life<=0){scene.remove(s);sparks.splice(i,1);} }
            powderPool.forEach(p=>{ if(p.active){ p.mesh.position.y+=p.vy; p.mesh.position.x+=p.vx; if(p.mesh.position.y<eYMax){p.active=false;p.mesh.visible=false;} } });

            if(window.viewMode === 'micro' || window.viewMode === 'interactive') {
                const auto = window.isPlaying && window.viewMode!=='interactive' && !isAnodeEffect;
                alIons.forEach(ion => {
                    if(ion.state === 'falling' && auto && ion.mesh !== draggedObj) {
                        ion.mesh.position.y -= ion.speed * window.powerT; if(ion.mesh.position.y < eYMin+0.8) handleDeposition(ion.mesh);
                    } else if(ion.state === 'receiving') {
                        ion.timer+=1*window.powerT; if(ion.timer > 40) { ion.state='deposited'; ion.timer=0; ion.mesh.material=matAlMetal; ion.mesh.userData.name="Al 铝单质(液态)"; }
                    } else if(ion.state === 'deposited') {
                        ion.timer+=1*window.powerT; ion.mesh.position.y-=0.01*window.powerT; ion.mesh.scale.x+=0.01*window.powerT; ion.mesh.scale.z+=0.01*window.powerT; ion.mesh.scale.y-=0.01*window.powerT;
                        if(ion.timer > 40) { 
                            ion.state='falling'; ion.mesh.material=matAl; ion.mesh.userData.name="Al³⁺"; ion.mesh.scale.setScalar(1); ion.mesh.position.set(THREE.MathUtils.randFloat(eXMin,eXMax), eYMax, THREE.MathUtils.randFloat(eZMin,eZMax)); 
                        }
                    }
                });
                oIons.forEach(ion => {
                    if(ion.state === 'rising' && auto && ion.mesh !== draggedObj) {
                        if(!ion.targetSite) { let a=anodeSites.filter(s=>s.count<2); if(a.length) ion.targetSite=a[Math.floor(Math.random()*a.length)]; }
                        if(ion.targetSite) {
                            let dx=ion.targetSite.x-ion.mesh.position.x, dy=ion.targetSite.y-ion.mesh.position.y, dz=ion.targetSite.z-ion.mesh.position.z, dist=Math.sqrt(dx*dx+dy*dy+dz*dz);
                            if(dist<0.1){ ion.state='binding'; ion.targetSite.count++; ion.targetSite.oxIons.push(ion); if(ion.targetSite.count===1){ion.mesh.position.set(ion.targetSite.x-0.15,ion.targetSite.y,ion.targetSite.z);} else if(ion.targetSite.count===2){ handleFusion(ion.targetSite.oxIons[0].mesh,ion.targetSite.oxIons[1].mesh); ion.targetSite.count=0; ion.targetSite.oxIons=[]; } }
                            else { ion.mesh.position.x+=(dx/dist)*ion.speed*window.powerT; ion.mesh.position.y+=(dy/dist)*ion.speed*window.powerT; ion.mesh.position.z+=(dz/dist)*ion.speed*window.powerT; }
                        } else { ion.mesh.position.y+=ion.speed*window.powerT; if(ion.mesh.position.y>eYMax) ion.mesh.position.y=eYMin; }
                    }
                });
                o2Pool.forEach(g => { if(g.active){ g.mesh.position.y+=g.speed*window.powerT; g.mesh.position.x+=Math.sin(Date.now()*0.005+g.mesh.position.y)*0.005*window.powerT; g.mesh.rotation.z+=0.05*window.powerT; g.mesh.rotation.y+=0.05*window.powerT; if(g.mesh.position.y>eYMax+2){g.active=false;g.mesh.visible=false;} } });
                for(let i=reactionElectrons.length-1; i>=0; i--) {
                    const e = reactionElectrons[i]; if(e.userData.target) { e.userData.progress+=e.userData.speed*window.powerT; e.position.lerp(e.userData.target.position, 0.1*window.powerT); } else { e.userData.progress+=0.02*window.powerT; e.position.add(e.userData.dir.clone().multiplyScalar(0.08*window.powerT)); }
                    if(e.userData.progress>=1) { particlesGroup.remove(e); reactionElectrons.splice(i,1); }
                }
                const bwn = (window.isPlaying||window.viewMode==='interactive')?window.powerT:0;
                [naIons, alf6Ions, alIons, oIons].forEach(arr=>arr.forEach(ion=>{ if(ion.state&&ion.state!=='rising'&&ion.state!=='falling')return; if(ion.mesh===draggedObj)return; (ion.mesh||ion).position.add(new THREE.Vector3(THREE.MathUtils.randFloatSpread(0.02),THREE.MathUtils.randFloatSpread(0.02),THREE.MathUtils.randFloatSpread(0.02)).multiplyScalar(bwn)); }));
            }

            const time = Date.now() * 0.005; const waveAmp = isAnodeEffect ? 0.2 : 0.08;
            for(let i=0; i<alPosAttr.count; i++) if(alPoolGeom.userData.baseY[i]>0.5) alPosAttr.setY(i, alPoolGeom.userData.baseY[i]+(Math.sin(alPosAttr.getX(i)*1.5+time)+Math.cos(alPosAttr.getZ(i)*1.5+time*0.8))*waveAmp*window.powerT);
            alPosAttr.needsUpdate = true;

            raycaster.setFromCamera(mouse, camera); const hits = raycaster.intersectObjects(scene.children, true);
            let found = null; for(let h of hits) { let o=h.object; let v=true; let c=o; while(c){if(!c.visible){v=false;break;}c=c.parent;} if(!v)continue; let n=o.userData.name||(o.parent&&o.parent.userData.name); if(n) { found=n; if(n!=="碳槽内壁(阴极)"&&n!=="碳槽侧壁(阴极)"&&n!=="碳槽底部(阴极)") break; } }
            if(window.powerT<0.2&&found&&window.viewMode!=='macro'&&window.viewMode!=='crystal'&&found.includes('离子')) found += " (冻结)";
            if(tooltip) { if(found&&!draggedObj&&window.viewMode!=='crystal'){tooltip.innerText=found;tooltip.classList.remove('hidden','opacity-0');} else {tooltip.classList.add('hidden','opacity-0');} }

            renderer.render(scene, camera);
        }
        animate();
