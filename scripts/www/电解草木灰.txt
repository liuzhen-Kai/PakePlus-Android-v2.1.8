window.onload = function() {
    // --- 核心状态与 UI DOM ---
    let mode = 'aqueous'; 
    let experimentActive = false;
    let isPaused = false;
    let isSlowMo = false;
    const speedScale = 0.3; 
    
    // 摄像机导播目标
    let targetCamPos = new THREE.Vector3(0, 15, 35);
    let targetLookAt = new THREE.Vector3(0, 3, 0);
    let isAutoCamera = false; // 新增：是否处于自动镜头过渡状态
    
    // 手电筒追踪变量
    let isTracking = false;
    let trackedIon = null;

    // 气体积
    let gasCathode = 0; let gasAnode = 0; const MAX_GAS = 100;

    const powerBtn = document.getElementById('powerBtn');
    const eqTitle = document.getElementById('eqTitle');
    const eqCathode = document.getElementById('eqCathode');
    const eqAnode = document.getElementById('eqAnode');
    const gasChart = document.getElementById('gasChart');
    const barH2 = document.getElementById('barH2');
    const barO2 = document.getElementById('barO2');
    const btnTrack = document.getElementById('btnTrack');
    const btnPause = document.getElementById('btnPause');
    const btnSlow = document.getElementById('btnSlow');

    // --- Three.js 初始化 ---
    const scene = new THREE.Scene();
    scene.fog = new THREE.FogExp2(0x0a0f16, 0.015);

    const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 1000);
    camera.position.copy(targetCamPos);

    const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    renderer.setPixelRatio(window.devicePixelRatio);
    renderer.shadowMap.enabled = true;
    document.getElementById('canvas3d').appendChild(renderer.domElement);

    const controls = new THREE.OrbitControls(camera, renderer.domElement);
    controls.enableDamping = true; controls.dampingFactor = 0.05;
    controls.target.copy(targetLookAt);

    // --- 新增：监听用户手动操作，打断自动镜头，恢复自由视角 ---
    controls.addEventListener('start', () => {
        isAutoCamera = false; // 用户一碰屏幕，就关闭自动镜头强制锁定
        if(isTracking) {
            isTracking = false;
            btnTrack.classList.remove('active-track');
        }
    });

    // --- 灯光系统 (包含手电筒追踪光) ---
    const ambientLight = new THREE.AmbientLight(0xffffff, 0.6);
    scene.add(ambientLight);
    const dirLight = new THREE.DirectionalLight(0xfff0dd, 0.8);
    dirLight.position.set(10, 20, 10); dirLight.castShadow = true;
    scene.add(dirLight);
    const backLight = new THREE.DirectionalLight(0xaabbff, 0.4);
    backLight.position.set(-10, 10, -10);
    scene.add(backLight);
    
    const purpleLight = new THREE.PointLight(0xcc44ff, 0, 15);
    purpleLight.position.set(-4, 2, 0);
    scene.add(purpleLight);

    // 追踪聚光灯 (默认隐藏)
    const trackLight = new THREE.SpotLight(0xffffff, 0, 20, Math.PI/6, 0.8, 1);
    trackLight.position.set(0, 10, 0);
    scene.add(trackLight);
    scene.add(trackLight.target);

    // --- 材质与分子构建 ---
    const materials = {
        glass: new THREE.MeshPhysicalMaterial({ color: 0xffffff, transmission: 0.9, opacity: 1, transparent: true, roughness: 0.05, ior: 1.5, side: THREE.DoubleSide, depthWrite: false }),
        waterBase: new THREE.MeshPhysicalMaterial({ color: 0x88ccff, transmission: 0, opacity: 0.35, transparent: true, roughness: 0.1, depthWrite: false }),
        waterTube: new THREE.MeshPhysicalMaterial({ color: 0x4488cc, transmission: 0, opacity: 0.55, transparent: true, roughness: 0.1, depthWrite: false }),
        dryAsh: new THREE.MeshStandardMaterial({ color: 0x8a7a62, roughness: 1.0 }),
        wetMud: new THREE.MeshStandardMaterial({ color: 0x5c4a37, roughness: 0.8, metalness: 0.1, transparent: true, opacity: 0.55 }),
        electrodeCathode: new THREE.MeshStandardMaterial({ color: 0x222222, metalness: 0.8, roughness: 0.3 }),
        electrodeAnode: new THREE.MeshStandardMaterial({ color: 0x222222, metalness: 0.8, roughness: 0.3 })
    };

    const cpkMats = {
        H: new THREE.MeshStandardMaterial({ color: 0xffffff, roughness: 0.1, metalness: 0.1 }),
        O: new THREE.MeshStandardMaterial({ color: 0xdd1111, roughness: 0.2, metalness: 0.1 }),
        C: new THREE.MeshStandardMaterial({ color: 0x222222, roughness: 0.4, metalness: 0.2 }),
        K: new THREE.MeshStandardMaterial({ color: 0x8a2be2, roughness: 0.2, metalness: 0.2 })
    };

    function buildMolecule(type) {
        const group = new THREE.Group();
        if (type === 'K') { group.add(new THREE.Mesh(cpkGeos.K, cpkMats.K)); }
        else if (type === 'H') { group.add(new THREE.Mesh(cpkGeos.H, cpkMats.H)); }
        else if (type === 'OH') { const o = new THREE.Mesh(cpkGeos.O, cpkMats.O); const h = new THREE.Mesh(cpkGeos.H, cpkMats.H); h.position.set(0, -0.16, 0); group.add(o, h); }
        else if (type === 'H2O') { const o = new THREE.Mesh(cpkGeos.O, cpkMats.O); const h1 = new THREE.Mesh(cpkGeos.H, cpkMats.H); h1.position.set(0.13, -0.1, 0); const h2 = new THREE.Mesh(cpkGeos.H, cpkMats.H); h2.position.set(-0.13, -0.1, 0); group.add(o, h1, h2); }
        else if (type === 'CO3') { const c = new THREE.Mesh(cpkGeos.C, cpkMats.C); const o1 = new THREE.Mesh(cpkGeos.O, cpkMats.O); o1.position.set(0, 0.2, 0); const o2 = new THREE.Mesh(cpkGeos.O, cpkMats.O); o2.position.set(0.17, -0.1, 0); const o3 = new THREE.Mesh(cpkGeos.O, cpkMats.O); o3.position.set(-0.17, -0.1, 0); group.add(c, o1, o2, o3); }
        else if (type === 'H2') { const h1 = new THREE.Mesh(cpkGeos.H, cpkMats.H); h1.position.set(0.09, 0, 0); const h2 = new THREE.Mesh(cpkGeos.H, cpkMats.H); h2.position.set(-0.09, 0, 0); group.add(h1, h2); }
        else if (type === 'O2') { const o1 = new THREE.Mesh(cpkGeos.O, cpkMats.O); o1.position.set(0.12, 0, 0); const o2 = new THREE.Mesh(cpkGeos.O, cpkMats.O); o2.position.set(-0.12, 0, 0); group.add(o1, o2); }
        return group;
    }

    // --- 场景搭建 ---
    const apparatus = new THREE.Group(); scene.add(apparatus);
    
    // 基础容器
    const table = new THREE.Mesh(new THREE.BoxGeometry(40, 1, 30), new THREE.MeshStandardMaterial({ color: 0x2a3036, roughness: 0.9 })); table.position.y = -0.5; apparatus.add(table);
    const tankWidth = 16, tankHeight = 8, tankDepth = 10;
    const tank = new THREE.Mesh(new THREE.BoxGeometry(tankWidth, tankHeight, tankDepth), materials.glass); tank.position.y = tankHeight / 2; apparatus.add(tank);
    const baseBorder = new THREE.Mesh(new THREE.BoxGeometry(tankWidth+0.4, 0.5, tankDepth+0.4), new THREE.MeshStandardMaterial({color:0x445566})); baseBorder.position.y = 0.25; apparatus.add(baseBorder);

    // 内容物 (严格分层渲染)
    const liquidMesh = new THREE.Mesh(new THREE.BoxGeometry(tankWidth-0.4, tankHeight-1, tankDepth-0.4), materials.waterBase); liquidMesh.position.y = 3.7; liquidMesh.renderOrder = 1; apparatus.add(liquidMesh);
    const ashMesh = new THREE.Mesh(new THREE.BoxGeometry(tankWidth-0.4, 3, tankDepth-0.4), materials.dryAsh); ashMesh.position.y = 1.7; apparatus.add(ashMesh);
    const mudMesh = new THREE.Mesh(new THREE.BoxGeometry(tankWidth-0.4, 2.5, tankDepth-0.4), materials.wetMud); mudMesh.position.y = 1.45; apparatus.add(mudMesh);

    // 电极 (包裹式)
    const elecGeo = new THREE.CylinderGeometry(0.4, 0.4, 8, 16);
    const cathode = new THREE.Mesh(elecGeo, materials.electrodeCathode); cathode.position.set(-4, 5, 0); apparatus.add(cathode);
    const anode = new THREE.Mesh(elecGeo, materials.electrodeAnode); anode.position.set(4, 5, 0); apparatus.add(anode);

    // 试管与内部水柱
    const tubeGroup = new THREE.Group(); apparatus.add(tubeGroup);
    const tubeGeo = new THREE.CylinderGeometry(1.2, 1.2, 8, 16, 1, true);
    const tubeTopGeo = new THREE.SphereGeometry(1.2, 16, 16, 0, Math.PI*2, 0, Math.PI/2);
    
    const tubeLeft = new THREE.Mesh(tubeGeo, materials.glass); tubeLeft.position.set(-4, 6, 0); tubeLeft.renderOrder = 3;
    const tubeLeftTop = new THREE.Mesh(tubeTopGeo, materials.glass); tubeLeftTop.position.set(0, 4, 0); tubeLeftTop.renderOrder = 3; tubeLeft.add(tubeLeftTop); tubeGroup.add(tubeLeft);
    const tubeRight = new THREE.Mesh(tubeGeo, materials.glass); tubeRight.position.set(4, 6, 0); tubeRight.renderOrder = 3;
    const tubeRightTop = new THREE.Mesh(tubeTopGeo, materials.glass); tubeRightTop.position.set(0, 4, 0); tubeRightTop.renderOrder = 3; tubeRight.add(tubeRightTop); tubeGroup.add(tubeRight);

    const innerWaterGeo = new THREE.CylinderGeometry(1.1, 1.1, 8, 16); innerWaterGeo.translate(0, 4, 0);
    const waterLeft = new THREE.Mesh(innerWaterGeo, materials.waterTube); waterLeft.position.set(-4, 2, 0); waterLeft.renderOrder = 2; tubeGroup.add(waterLeft);
    const waterRight = new THREE.Mesh(innerWaterGeo, materials.waterTube); waterRight.position.set(4, 2, 0); waterRight.renderOrder = 2; tubeGroup.add(waterRight);

    // 悬浮式仪表盘 HUD 电源
    const psGroup = new THREE.Group(); psGroup.position.set(0, 14, 0); apparatus.add(psGroup);
    const psBody = new THREE.Mesh(new THREE.BoxGeometry(9, 3.5, 3.5), new THREE.MeshStandardMaterial({ color: 0x2c3e50, metalness: 0.4, roughness: 0.7 })); psGroup.add(psBody);

    function initIonsForMode() {
        if(mode === 'dry') return;
        let activeCount = mode === 'water' ? 50 : 80; if(mode === 'wet') activeCount = 30;
        let i = 0;
        ions.forEach(p => {
            if(i < activeCount) {
                p.active = true;
                Object.values(p.models).forEach(m => m.visible = false);
                const r = Math.random();
                if (mode === 'aqueous') { if (r<0.3) p.type='K'; else if (r<0.5) p.type='H'; else if (r<0.7) p.type='OH'; else if (r<0.9) p.type='CO3'; else p.type='H2O'; }
                else if (mode === 'wet') { if (r<0.5) p.type='K'; else if (r<0.8) p.type='CO3'; else if (r<0.9) p.type='H2O'; else if (r<0.95) p.type='OH'; else p.type='H'; }
                else if (mode === 'water') { if (r<0.4) p.type='H'; else if (r<0.8) p.type='OH'; else p.type='H2O'; }
                p.models[p.type].visible = true;
                p.rvx = (Math.random()-0.5)*0.1; p.rvy = (Math.random()-0.5)*0.1; p.rvz = (Math.random()-0.5)*0.1;
                const rangeY = mode === 'wet' ? 2 : 5; const startY = mode === 'wet' ? 1.5 : 2;
                p.x = (Math.random() - 0.5) * 14; p.y = startY + Math.random() * rangeY; p.z = (Math.random() - 0.5) * 8;
                p.vx = (Math.random()-0.5)*0.02; p.vy = (Math.random()-0.5)*0.02; p.vz = (Math.random()-0.5)*0.02;
            } else p.active = false;
            i++;
        });
    }

    function spawnBubble(side) {
        const b = bubbles.find(bub => !bub.active);
        if(b) {
            b.active = true; b.side = side; b.type = side === -1 ? 'H2' : 'O2';
            Object.values(b.models).forEach(m => m.visible = false); b.models[b.type].visible = true;
            const r = 0.4 + Math.random() * 0.1; const angle = Math.random() * Math.PI * 2;
            b.x = (side===-1 ? -4 : 4) + Math.cos(angle) * r; b.z = Math.sin(angle) * r;
            if(mode === 'water') { const waterTop = 2 + 8 * (1 - (side===-1 ? gasCathode : gasAnode) / MAX_GAS); b.y = 1 + Math.random() * (Math.min(9, waterTop) - 1); }
            else { b.y = 1 + Math.random() * 8; }
            b.vy = 0.05 + Math.random() * 0.05;
            b.rvx = (Math.random()-0.5)*0.2; b.rvy = (Math.random()-0.5)*0.2; b.rvz = (Math.random()-0.5)*0.2;
        }
    }

    function spawnPotassiumSpark() {
        const s = potassiumSparks.find(sp => !sp.active);
        if(s) {
            s.active = true; s.age = 0; s.maxAge = 40 + Math.random()*20;
            const angle = Math.random() * Math.PI * 2;
            s.mesh.position.set(-4 + Math.cos(angle)*0.5, 2.5 + Math.random()*0.5, Math.sin(angle)*0.5);
            purpleLight.intensity = 1.5;
        }
    }

        }
    }

    // --- 动画主循环 ---
    function animate() {
        requestAnimationFrame(animate);
        
        // 1. 镜头补间动画 (Lerp) - 修复：仅在开启自动镜头时才强制锁定！
        if (isAutoCamera) {
            camera.position.lerp(targetCamPos, 0.08);
            controls.target.lerp(targetLookAt, 0.08);
        }
        controls.update();

        // 2. 微观追踪逻辑
        if(isTracking && trackedIon && trackedIon.active) {
            // 在追踪模式下，实时更新自动镜头的目标点
            targetLookAt.copy(trackedIon.group.position);
            targetCamPos.set(trackedIon.x, trackedIon.y + 4, trackedIon.z + 8);
            trackLight.intensity = 2;
            trackLight.position.set(trackedIon.x, trackedIon.y + 6, trackedIon.z);
            trackLight.target.position.copy(trackedIon.group.position);
            ambientLight.intensity = 0.1; dirLight.intensity = 0.1; // 压暗环境
        } else {
            trackLight.intensity = 0;
            ambientLight.intensity = 0.6; dirLight.intensity = 0.8;
            if(isTracking && experimentActive) { trackedIon = ions.find(i => i.active); } // 离子消失后换一个
        }

        // 3. 计算时间因子
        const dt = isPaused ? 0 : (isSlowMo ? 0.1 : 1.0);
        const curSpeedScale = speedScale * dt;

        // 4. 电表与电子流
        if (experimentActive && !isPaused) {
            let baseC = 0; if(mode==='aqueous') baseC=2.45; else if(mode==='wet') baseC=0.65; else if(mode==='water') baseC=1.95;
            if (baseC > 0) {
                updateAmmeter(baseC + (Math.random()-0.5)*0.04, true);
                if (Math.random() < baseC * 0.1 * curSpeedScale) { spawnElectron('neg'); spawnElectron('pos'); }
            }
        }
        
        electronPulses.forEach(e => {
            if(!e.active) { e.mesh.visible = false; return; }
            e.mesh.visible = true;
            if(dt > 0) { e.t += (e.type === 'neg' ? 0.03*dt : -0.03*dt); }
            if(e.t >= 1 || e.t <= 0) e.active = false;
            if(e.active && e.curve) e.mesh.position.copy(e.curve.getPointAt(e.t));
        });

        // 5. 离子运动物理引擎
        ions.forEach(p => {
            if(!p.active) { p.group.visible = false; return; }
            p.group.visible = true;
            
            if (experimentActive && mode !== 'dry' && dt > 0) {
                const E_field = 0.015 * curSpeedScale;
                if (p.type === 'K' || p.type === 'H') p.vx -= E_field;
                else if (p.type === 'OH' || p.type === 'CO3') p.vx += E_field;

                p.vx = p.vx * 0.95 + (Math.random()-0.5)*0.01 * curSpeedScale;
                p.vy = p.vy * 0.95 + (Math.random()-0.5)*0.01 * curSpeedScale;
                p.vz = p.vz * 0.95 + (Math.random()-0.5)*0.01 * curSpeedScale;

                p.x += p.vx; p.y += p.vy; p.z += p.vz;

                const limitX = 7, limitZ = 4.5; let limitYMin = 2, limitYMax = 7; if(mode === 'wet') { limitYMin = 1.5; limitYMax = 3.5; }
                if (p.x < -limitX) { p.x = -limitX; p.vx *= -0.5; } if (p.x > limitX) { p.x = limitX; p.vx *= -0.5; }
                if (p.y < limitYMin) { p.y = limitYMin; p.vy *= -0.5; } if (p.y > limitYMax) { p.y = limitYMax; p.vy *= -0.5; }
                if (p.z < -limitZ) { p.z = -limitZ; p.vz *= -0.5; } if (p.z > limitZ) { p.z = limitZ; p.vz *= -0.5; }

                if ((p.type === 'H' || p.type === 'K') && p.x < -3.5) p.x = 2 + Math.random()*4;
                if ((p.type === 'OH' || p.type === 'CO3') && p.x > 3.5) p.x = -2 - Math.random()*4;
            }
            
            p.group.position.set(p.x, p.y, p.z);
            if(dt > 0) { p.group.rotation.x += p.rvx * dt; p.group.rotation.y += p.rvy * dt; p.group.rotation.z += p.rvz * dt; }
        });

        // 6. 宏观反应现象 (气泡、降液面、燃烧)
        if (experimentActive && mode !== 'dry' && dt > 0) {
            if(mode === 'aqueous' || mode === 'water') {
                if(Math.random() < 0.4 * dt) spawnBubble(-1);
                if(Math.random() < 0.2 * dt) spawnBubble(1);
            } else if (mode === 'wet') {
                if(Math.random() < 0.05 * dt) spawnBubble(1);
                if(Math.random() < 0.08 * dt) spawnPotassiumSpark();
                purpleLight.intensity *= 0.95;
            }

            if(mode === 'water') {
                if(gasCathode < MAX_GAS) gasCathode += 0.2 * curSpeedScale;
                if(gasAnode < MAX_GAS) gasAnode += 0.1 * curSpeedScale;
                waterLeft.scale.y += (Math.max(0.01, 1 - gasCathode/MAX_GAS) - waterLeft.scale.y) * 0.1;
                waterRight.scale.y += (Math.max(0.01, 1 - gasAnode/MAX_GAS) - waterRight.scale.y) * 0.1;
                updateChartDOM();
            }
        }

        bubbles.forEach(b => {
            if(!b.active) { b.group.visible = false; return; }
            b.group.visible = true;
            if(dt > 0) {
                b.y += b.vy * dt;
                b.x += Math.sin(b.y * 10) * 0.01 * curSpeedScale;
                let topLimit = mode==='water' ? 2 + 8 * (1 - (b.side===-1 ? gasCathode : gasAnode) / MAX_GAS) : 7;
                if(b.y > topLimit) b.active = false;
                b.group.rotation.x += b.rvx * dt; b.group.rotation.y += b.rvy * dt; b.group.rotation.z += b.rvz * dt;
            }
            b.group.position.set(b.x, b.y, b.z);
        });

        potassiumSparks.forEach(s => {
            if(!s.active) { s.mesh.visible = false; return; }
            s.mesh.visible = true;
            if(dt > 0) {
                s.age += curSpeedScale; s.mesh.position.y += 0.02 * curSpeedScale;
                const scale = 0.5 * Math.sin((s.age/s.maxAge) * Math.PI); s.mesh.scale.set(scale+0.5, scale+0.5, scale+0.5);
                s.mesh.material.opacity = 1 - (s.age/s.maxAge);
                if(s.age >= s.maxAge) s.active = false;
            }
        });

        renderer.render(scene, camera);
    }

    setMode('aqueous');
    animate();

    window.addEventListener('resize', () => { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); });
};
