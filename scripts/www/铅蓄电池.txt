            const btnA = document.getElementById('btn-aging');
            
            if (chargeState.level <= 0 && chargeState.action === 'idle') return;

                if (currentTween) currentTween.kill();

                currentTween.pause();
                chargeState.action = 'paused_discharging';
                btnD.innerText = '继续放电';
                btnC.disabled = false; 
            } else if (chargeState.action === 'paused_discharging') {
                currentTween.resume();
                chargeState.action = 'discharging';
                btnD.innerText = '暂停放电';
                btnC.disabled = true;

                chargeState.action = 'discharging';
                btnD.innerText = '暂停放电';

                btnA.disabled = true;
                btnC.innerText = '充电';

            macroMat.posPlate.color.lerpColors(colors.posActive, colors.discharged, global_rp);
            macroMat.negPlate.color.lerpColors(colors.negActive, colors.discharged, global_rp);

            if (chargeState.action.includes('discharging') || chargeState.action.includes('charging')) {
                electrons.forEach((e, idx) => {
                    let phase = (chargeState.time * 3 + idx * 0.3) % 1.0; 
                    if (chargeState.action.includes('charging')) phase = 1.0 - phase; 
                    if (e.type === 'neg') {
                        e.mesh.position.set(e.x, e.startY + (e.targetY - e.startY) * phase, e.z + Math.sin(phase*20)*0.01);
                    } else {
                        e.mesh.position.set(e.x, e.startY + (e.targetY - e.startY) * phase, e.z + Math.cos(phase*20)*0.01);
                    }
                });
            }

            const overlap = 0.8; 
            activeSitesPos.forEach((site, i) => {
                const orderIndex = site.layer * (ROWS * COLS) + (i % (ROWS * COLS));
                const start = (orderIndex / NUM_SITES) * (1 - overlap);
                const end = start + overlap;
                let local_rp = Math.max(0, Math.min(1, (global_rp - start) / (end - start)));
                
                // ⚠️ 老化死晶体锁死逻辑
                if(site.isDead) {
                    local_rp = 1.0; 
                    site.pb.material = deadMat;
                    site.pb.scale.setScalar(0.06); 
                    site.so4.children.forEach(c => { if(c.material) c.material = deadMat; });
                }
                
                site.local_rp = local_rp; 

                let p1 = Math.min(Math.max(local_rp / 0.3, 0), 1);         
                let p2 = Math.min(Math.max((local_rp - 0.3) / 0.3, 0), 1);   
                let p3 = Math.min(Math.max((local_rp - 0.6) / 0.4, 0), 1);   

                // 添加物理热震荡
                let jitterX = Math.sin(time * 40 + i) * thermalJitter;
                let jitterY = Math.cos(time * 50 + i) * thermalJitter;

                site.pb.position.lerpVectors(site.pbOrig, site.pbDest, Math.max(p2, p3)); 
                site.pb.position.x += jitterX; site.pb.position.y += jitterY;

                let w1Center = new THREE.Vector3().lerpVectors(site.o1Start, site.idleWater1, p2);
                let w2Center = new THREE.Vector3().lerpVectors(site.o2Start, site.idleWater2, p2);
                if (p2 > 0) {
                    let drift = new THREE.Vector3(Math.sin(time+site.rotOffset)*0.05, Math.cos(time)*0.05, 0);
                    w1Center.add(drift.clone().multiplyScalar(p2)); w2Center.sub(drift.clone().multiplyScalar(p2));
                }
                
                site.o1.position.copy(w1Center); site.o2.position.copy(w2Center);
                site.o1.position.x += jitterX; site.o2.position.y += jitterY;

                if (p2 === 0) {
                    site.h1.position.lerpVectors(site.idleH1, site.dockH1, p1); site.h2.position.lerpVectors(site.idleH2, site.dockH2, p1);
                    site.h3.position.lerpVectors(site.idleH3, site.dockH3, p1); site.h4.position.lerpVectors(site.idleH4, site.dockH4, p1);
                } else {
                    let rot1 = time * 0.5 + site.rotOffset;
                    site.h1.position.set(w1Center.x + Math.cos(rot1)*0.02, w1Center.y + Math.sin(rot1)*0.015, w1Center.z);
                    site.h2.position.set(w1Center.x + Math.cos(rot1+2)*0.02, w1Center.y + Math.sin(rot1+2)*0.015, w1Center.z);
                    let rot2 = -time * 0.5 - site.rotOffset;
                    site.h3.position.set(w2Center.x + Math.cos(rot2)*0.02, w2Center.y + Math.sin(rot2)*0.015, w2Center.z);
                    site.h4.position.set(w2Center.x + Math.cos(rot2+2)*0.02, w2Center.y + Math.sin(rot2+2)*0.015, w2Center.z);
                }

                let currentSO4 = new THREE.Vector3();
                if (p3 === 0) {
                    currentSO4.lerpVectors(site.idleSO4, site.waitSO4, p1); 
                    site.so4.rotation.set(time*0.5+site.rotOffset, time*0.5+site.rotOffset, 0);
                } else {
                    currentSO4.lerpVectors(site.waitSO4, site.dockSO4, p3);
                    qIdle.setFromEuler(new THREE.Euler(time*0.5+site.rotOffset, time*0.5+site.rotOffset, 0));
                    qDock.setFromEuler(eDockL);
                    site.so4.quaternion.slerpQuaternions(qIdle, qDock, p3);
                }
                site.so4.position.copy(currentSO4);
                site.so4.position.y += jitterY;
            });

            activeSitesNeg.forEach((site, i) => {
                const orderIndex = site.layer * (ROWS * COLS) + (i % (ROWS * COLS));
                const start = (orderIndex / NUM_SITES) * (1 - overlap);
                const end = start + overlap;
                let local_rp = Math.max(0, Math.min(1, (global_rp - start) / (end - start)));
                
                if (site.isDead) local_rp = 1.0; 

                let p1 = Math.min(Math.max(local_rp / 0.3, 0), 1);
                let p2 = Math.min(Math.max((local_rp - 0.3) / 0.3, 0), 1); 
                let p3 = Math.min(Math.max((local_rp - 0.6) / 0.4, 0), 1);

                let jitterX = Math.sin(time * 40 + i + 10) * thermalJitter;
                let jitterY = Math.cos(time * 50 + i + 10) * thermalJitter;

                site.pb.position.lerpVectors(site.pbOrig, site.pbDest, Math.max(p2, p3)); 
                site.pb.position.x += jitterX; site.pb.position.y += jitterY;

                if(site.isDead && site.pb.material !== deadMat) {
                    site.pb.material = deadMat; 
                    site.pb.scale.setScalar(0.06); 
                    site.so4.children.forEach(c => { if(c.material) c.material = deadMat; });
                }

                let currentSO4 = new THREE.Vector3();
                if (p3 === 0) {
                    currentSO4.lerpVectors(site.idleSO4, site.waitSO4, Math.max(p1, p2)); 
                    site.so4.rotation.set(time*0.5+site.rotOffset, time*0.5+site.rotOffset, 0);
                } else {
                    currentSO4.lerpVectors(site.waitSO4, site.dockSO4, p3);
                    qIdle.setFromEuler(new THREE.Euler(time*0.5+site.rotOffset, time*0.5+site.rotOffset, 0));
                    qDock.setFromEuler(eDockR);
                    site.so4.quaternion.slerpQuaternions(qIdle, qDock, p3);
                }
                site.so4.position.copy(currentSO4);
                site.so4.position.y += jitterY;
            });

            bgIons.forEach(ion => {
                if(ion.type === 'so4') {
                    ion.mesh.position.set(ion.pos.x + Math.sin(time+ion.offset)*0.05, ion.pos.y + Math.cos(time*0.8+ion.offset)*0.05, ion.pos.z + Math.sin(time*0.9+ion.offset)*0.05);
                    ion.mesh.rotation.set(time*0.4, time*0.4, 0);
                } else {
                    ion.mesh.position.set(ion.pos.x + Math.sin(time*1.2+ion.offset)*0.06, ion.pos.y + Math.cos(time*1.1+ion.offset)*0.06, ion.pos.z + Math.sin(time*1.3+ion.offset)*0.06);
                }
            });

            if (currentView === 'macro' && !isViewAnimating) {
                raycaster.setFromCamera(mouse, camera);
                let allMeshes = [];
                [posGroup, negGroup, liqGroup].forEach(g => {
                    g.children.forEach(c => allMeshes.push(c));
                });
                const intersects = raycaster.intersectObjects(allMeshes, false);
                
                if (intersects.length > 0 && intersects[0].object.userData.tooltip) {
                    tooltip.innerHTML = intersects[0].object.userData.tooltip;
                    tooltip.style.left = (mousePx.x + 15) + 'px';
                    tooltip.style.top = (mousePx.y + 15) + 'px';
                    tooltip.classList.add('visible');
                } else {
                    tooltip.classList.remove('visible');
                }

                // 拆解标签动态跟踪
                const expVal = parseFloat(explodeSlider.value);
                if (expVal > 0.05) {
                    const wHalf = window.innerWidth / 2, hHalf = window.innerHeight / 2;
                    const updateLabel = (id, obj, ox, oy, oz) => {
                        const el = document.getElementById(id);
                        if(el) {
                            let vec = new THREE.Vector3(ox, oy, oz).applyMatrix4(obj.matrixWorld).project(camera);
                            if (vec.z < 1) {
                                el.style.left = `${(vec.x * wHalf) + wHalf}px`; el.style.top = `${-(vec.y * hHalf) + hHalf}px`;
                                el.classList.add('visible');
                            } else { el.classList.remove('visible'); }
                        }
                    };
                    updateLabel('label-ex-pos', posGroup, 2.1, 2.5, 2.5);
                    updateLabel('label-ex-neg', negGroup, -3.1, 2.5, -2.5);
                    updateLabel('label-ex-liq', liqGroup, 0, 1.0, 3.5);
                    updateLabel('label-ex-cas', casingGroup, 0, -3.5, 3.5);
                } else {
                    ['label-ex-pos', 'label-ex-neg', 'label-ex-liq', 'label-ex-cas'].forEach(id => {
                        let el = document.getElementById(id);
                        if(el) el.classList.remove('visible');
                    });
                }
            } else {
                tooltip.classList.remove('visible');
                ['label-ex-pos', 'label-ex-neg', 'label-ex-liq', 'label-ex-cas'].forEach(id => {
                    let el = document.getElementById(id);
                    if(el) el.classList.remove('visible');
                });
            }

            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        camera.position.set(12, 10, 15); controls.target.set(0, -1, 0);
        applyAlphaRules();
        switchView('macro');
        animate();
    </script>
