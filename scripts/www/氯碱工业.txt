        const clock = new THREE.Clock();
        function animate() {
            requestAnimationFrame(animate);
            const delta = clock.getDelta();
            
            TWEEN.update();
            controls.update();

            if (isPlaying) {
                reactionProgress += delta * 0.05; 
                if (reactionProgress > 1) reactionProgress = 1;
                
                if (!isConcreteView) {
                    if (!isFaultMode) {
                        materials.anodeLiquid.color.lerpColors(new THREE.Color(0x4da6ff), new THREE.Color(0xccffff), reactionProgress * 0.1);
                        materials.cathodeLiquid.color.lerpColors(new THREE.Color(0x9933cc), new THREE.Color(0x5e1b82), reactionProgress * 0.1);
                        materials.cathodeLiquid.opacity = 0.3 + (reactionProgress * 0.5); 
                    } else {
                        materials.anodeLiquid.color.lerpColors(new THREE.Color(0x4da6ff), new THREE.Color(0x99cc33), reactionProgress * 0.1);
                    }
                }

                const time = Date.now() * 0.005;
                flowArrows.forEach(arrow => {
                    if(arrow.userData.directionStr === 'up') arrow.position.y = arrow.userData.origY + Math.sin(time) * 0.1;
                    else if (arrow.userData.directionStr === 'left') arrow.position.x = arrow.userData.origX - Math.sin(time) * 0.1;
                    else if (arrow.userData.directionStr === 'right') arrow.position.x = arrow.userData.origX + Math.sin(time) * 0.1;
                });
            }

            electronParticles.forEach(e => {
                if (isPlaying) {
                    e.progress += delta * 0.4;
                    if (e.progress > 1) e.progress = 0;
                    e.mesh.position.copy(e.curve.getPointAt(e.progress));
                }
            });

            particles.forEach(p => {
                const floatOffset = Math.sin(Date.now() * 0.002 + p.originalPos.x) * 0.1; 

                if (p.type === 'cl2_gas' || p.type === 'h2_gas') {
                    if (isPlaying) {
                        p.mesh.position.y += delta * 1.5;
                        if (p.mesh.position.y > 4.5) p.mesh.position.y = 1.5; 
                    }
                } else if (p.type === 'na_ion') {
                    if (isPlaying) {
                        p.activeTime += delta;
                        if (p.activeTime > p.delay) {
                            p.progress += delta * p.speed;
                            if (p.progress > 1) {
                                p.progress = 0; p.activeTime = 0; p.mesh.position.copy(p.originalPos);
                            } else {
                                const targetX = 3.5;
                                p.mesh.position.x = p.originalPos.x + (p.progress * (targetX - p.originalPos.x));
                                p.mesh.position.y = p.originalPos.y + Math.sin(p.progress * Math.PI) * p.curveY;
                                p.mesh.position.z = p.originalPos.z + Math.sin(p.progress * Math.PI) * p.curveZ;
                            }
                        } else {
                            p.mesh.position.y = p.originalPos.y + floatOffset;
                        }
                    } else {
                        p.mesh.position.y = p.originalPos.y + floatOffset;
                    }
                } else if (p.type === 'cl_ion' || p.type === 'h_ion') {
                    if (isPlaying) {
                        p.mesh.position.x += (p.type === 'cl_ion' ? -1 : 1) * delta * p.speed;
                        if (p.type === 'cl_ion' && p.mesh.position.x < -4.0) p.mesh.position.x = -1.5 + Math.random();
                        if (p.type === 'h_ion' && p.mesh.position.x > 4.0) p.mesh.position.x = 1.5 - Math.random();
                    }
                    p.mesh.position.y = p.originalPos.y + floatOffset;
                } else if (p.type === 'oh_ion') {
                    if (isPlaying && isFaultMode) {
                        p.mesh.position.x -= delta * p.speed;
                        if (p.mesh.position.x < -4.0) p.mesh.position.x = 3.5; 
                    } else {
                        p.mesh.position.y = p.originalPos.y + floatOffset;
                    }
                }
            });

            updateLabels();
            renderer.render(scene, camera);
        }

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }, false);

        animate();
    </script>
